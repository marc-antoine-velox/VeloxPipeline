function _buildGroupingKey(params) {
  const today   = Utilities.formatDate(new Date(), 'America/Toronto', 'yyyy-MM-dd');
  const dealer  = (params.dealer || '').replace(/[^a-zA-Z0-9]/g, '-');
  const subject = (params.email_subject || '')
    .replace(/\s*\[AUTO[^\]]*\]/g, '')
    .replace(/[^a-zA-Z0-9À-ÿ\s]/g, '-')
    .trim()
    .substring(0, 60);
  return today + '_' + dealer + '_' + subject;
}

function normalizeDealer(extractedDealer, knownDealers) {
  if (!extractedDealer || extractedDealer.trim().length === 0) return extractedDealer;
  if (!knownDealers || knownDealers.length === 0) return extractedDealer;

  const input = extractedDealer.trim().toUpperCase();

  // Exact match first — handles Battelec vs Battelec Canada correctly
  const exactMatch = knownDealers.find(function(d) {
    return d.toUpperCase() === input;
  });
  if (exactMatch) return exactMatch;

  // Explicit alias — Hansler variants → Leavitt Machinery
  const ALIASES = {
    'HANSLER':              'Leavitt Machinery',
    'HANSLER/LEAVITT':      'Leavitt Machinery',
    'HANSLER LEAVITT':      'Leavitt Machinery',
    'ELEX':                 'Elevex',
    'TOROMONT MANUTENTION': 'Toromont',
    'TOROMONT CAT':         'Toromont'
  };
  if (ALIASES[input]) {
    Logger.log('Dealer alias applied: "' + extractedDealer + '" → "' + ALIASES[input] + '"');
    return ALIASES[input];
  }

  // Fuzzy match via Levenshtein similarity
  let bestMatch = null;
  let bestScore = 0;

  knownDealers.forEach(function(known) {
    const score = _stringSimilarity(input, known.toUpperCase());
    if (score > bestScore) {
      bestScore = score;
      bestMatch = known;
    }
  });

  if (bestScore >= 0.75 && bestMatch) {
    if (extractedDealer !== bestMatch) {
      Logger.log('Dealer normalized: "' + extractedDealer + '" → "' + bestMatch +
                 '" (similarity: ' + Math.round(bestScore * 100) + '%)');
    }
    return bestMatch;
  }

  Logger.log('Dealer kept as-is: "' + extractedDealer +
             '" — best match "' + (bestMatch || 'none') +
             '" at ' + Math.round(bestScore * 100) + '%');
  return extractedDealer;
}

function _stringSimilarity(a, b) {
  const la = a.length;
  const lb = b.length;
  if (la === 0 || lb === 0) return 0;
  const matrix = [];
  for (var i = 0; i <= lb; i++) matrix[i] = [i];
  for (var j = 0; j <= la; j++) matrix[0][j] = j;
  for (var i = 1; i <= lb; i++) {
    for (var j = 1; j <= la; j++) {
      const cost = a[j-1] === b[i-1] ? 0 : 1;
      matrix[i][j] = Math.min(
        matrix[i-1][j] + 1,
        matrix[i][j-1] + 1,
        matrix[i-1][j-1] + cost
      );
    }
  }
  return 1 - (matrix[lb][la] / Math.max(la, lb));
}

function writeSimulationToSheet(params) {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const dataSheet = ss.getSheetByName('Data Entry');
  dataSheet.getRange('J4').setValue(params.battery_ah);
  dataSheet.getRange('J5').setValue(params.truck_load);
  dataSheet.getRange('J6').setValue(params.truck_efficiency);
  dataSheet.getRange('J7').setValue(params.charger_amps);
  dataSheet.getRange('J8').setValue(0.80);
  dataSheet.getRange('J9').setValue(100);
  dataSheet.getRange('J10').setValue(params.shift1_label);
  dataSheet.getRange('J11').setValue(params.shift2_label);
  dataSheet.getRange('J12').setValue(params.shift3_label);
  dataSheet.getRange('J13').setValue(24);
  dataSheet.getRange('J14').setValue(1);
  dataSheet.getRange('J15').setValue(20);
  dataSheet.getRange('J16').setValue('12:00 AM');
  dataSheet.getRange('B4:G13').clearContent();
  const events = params.charging_events;
  for (let i = 0; i < events.length; i++) {
    const row = 4 + i;
    dataSheet.getRange(row, 2).setValue(i + 1);
    dataSheet.getRange(row, 3).setValue(events[i].start_day);
    dataSheet.getRange(row, 4).setValue(events[i].start_time);
    dataSheet.getRange(row, 5).setValue(events[i].end_day);
    dataSheet.getRange(row, 6).setValue(events[i].end_time);
    dataSheet.getRange(row, 7).setValue(events[i].label);
  }
  Utilities.sleep(2000);
  Logger.log('Simulation written to sheet');
}

function readSimulationResults() {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const dataSheet = ss.getSheetByName('Data Entry');
  const results = {
    total_ah:         dataSheet.getRange('M4').getValue(),
    allowed_ah:       dataSheet.getRange('M5').getValue(),
    remaining_ah:     dataSheet.getRange('M6').getValue(),
    status:           dataSheet.getRange('M7').getValue(),

  };
  Logger.log('Total Ah: ' + results.total_ah + ' | Status: ' + results.status + ' | Used%: ' + results.battery_used_pct);
  return results;
}

function logSimulationResult(params, results, simNumber) {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const logSheet = ss.getSheetByName('LOGS');

  // ── Deduplication — skip if identical run already logged for this thread ──
  const threadId = params.gmail_thread_id || '';
  if (threadId) {
    const existingData = logSheet.getDataRange().getValues();
    const heads = existingData[0];
    const threadCol   = heads.indexOf('Thread Gmail');
    const simCol      = heads.indexOf('No. Simulation');
    const statusCol   = heads.indexOf('Statut');
    const battCol     = heads.indexOf('Ah Lithium');
    const ampCol      = heads.indexOf('Ampères');
    const modelCol2   = heads.indexOf('Modèle');
    const scenarioCol = heads.indexOf('Scénario');

    let status;
    if (params.overall_passed === false) {
      status = 'FAIL';
    } else if (results.status === 'PASS' && results.battery_used_pct > 80) {
      status = 'REVIEW';
    } else if (results.status === 'PASS') {
      status = 'PASS';
    } else {
      status = 'FAIL';
    }

    for (var i = existingData.length - 1; i >= 1; i--) {
      const row = existingData[i];
      if (String(row[threadCol])   === threadId &&
          Number(row[simCol])      === simNumber &&
          String(row[statusCol])   === status &&
          Number(row[battCol])     === (params.battery_ah || 0) &&
          Number(row[ampCol])      === (params.charger_amps || 0) &&
          String(row[modelCol2])   === (params.forklift_model || '') &&
          String(row[scenarioCol]) === (params.scenario || 'STANDARD')) {
        Logger.log('LOGS: duplicate skipped — thread ' + threadId + ' sim ' + simNumber);
        return;
      }
    }
  }

  const nextRow = logSheet.getLastRow() + 1;

  let status;
  if (params.overall_passed === false) {
    status = 'FAIL';
  } else if (results.status === 'PASS' && results.battery_used_pct > 80) {
    status = 'REVIEW';
  } else if (results.status === 'PASS') {
    status = 'PASS';
  } else {
    status = 'FAIL';
  }

  let notes = params.notes || '';
  if (params.flags && params.flags.length > 0) notes += ' | ' + params.flags.join(' | ');

  const row = [
    new Date(),                               // A - Date/Heure
    params.dealer || '',                      // B - Dealer
    params.customer || '',                    // C - Client
    params.sender || '',                      // D - Expéditeur
    params.forklift_manufacturer || '',       // E - Fabricant
    params.forklift_model || '',              // F - Modèle
    params.voltage || '',                     // G - Voltage
    params.compartment || '',                 // H - Compartment
    params.battery_type || 'VeloxFlex',       // I - Battery_Type  ← NEW
    params.charger_model || '',               // J - Charger_Model ← NEW
    params.battery_ah || '',                  // K - Ah Lithium
    params.charger_amps || '',                // L - Ampères
    params.shifts || '',                      // M - Shifts
    params.work_hours || '',                  // N - Hrs Travail/Shift
    results.min_soc || '',                    // O - SOC Min
    status,                                   // P - Statut
    params.use_type || 'average',             // Q - Use Type
    params.lead_acid_ah || '',                // R - Ah Réf. LA
    params.truck_load || '',                  // S - Truck Load A/hm
    params.truck_efficiency || '',            // T - Efficacité %
    simNumber || 1,                           // U - No. Simulation
    params.heating_option ? 'OUI' : 'NON',   // V - Chauffage
    params.ip54_charger ? 'OUI' : 'NON',     // W - IP54
    params.gmail_thread_id || '',             // X - Thread Gmail
    params.forklift_class || '',              // Y - Classe Chariot
    params.special_equipment || '',           // Z - Équip. Spécial
    notes.trim(),                             // AA - Notes/Flags
    params.email_subject || '',               // AB - Sujet Email
    params.scenario || 'STANDARD',            // AC - Scénario
    params.language || '',                    // AD - Langue courriel
    '',                                       // AE - (reserved)
    _buildGroupingKey(params)                 // AF - Grouping Key
  ];

  logSheet.getRange(nextRow, 1, 1, 32).setValues([row]);
  Logger.log('Logged row ' + nextRow + ' | Sim ' + (simNumber || 1) + ' | ' + (params.scenario || 'STANDARD'));
}

function parseCompartmentInput(compartment) {
  if (!compartment) return { width: null, length: null, height: null };
  const parts = String(compartment).split(/[xX×]/).map(p => parseFloat(p.trim()));
  if (parts.length === 3 && !isNaN(parts[0]) && !isNaN(parts[1]) && !isNaN(parts[2])) {
    return { length: parts[0], width: parts[1], height: parts[2] };
  }
  if (parts.length === 1 && !isNaN(parts[0])) {
    return { width: parts[0], length: null, height: null };
  }
  return { width: null, length: null, height: null };
}

function lookupDekaPull(manufacturer, model, voltage, params, preloadedData) {
  const DEKA_SHEET_ID = '1HWGd1NCsbMFy9fzAgeC4LjU9kIdmKdAItG2hg-rTHV0';
  const DEKA_TAB = 'DATA';
  try {
    const data = preloadedData || SpreadsheetApp.openById(DEKA_SHEET_ID).getSheetByName(DEKA_TAB).getDataRange().getValues();
    const headers = data[0];
    const mfgCol         = headers.indexOf('Manufacturer');
    const modelCol       = headers.indexOf('Model');
    const voltCol        = headers.indexOf('Voltage');
    const compartmentCol = headers.indexOf('Compartment');
    const ahCol          = headers.indexOf('AhCapacity');
    const plateRatingCol = headers.indexOf('PlateRatingAh');
    Logger.log('Column indexes — Mfg:' + mfgCol + ' Model:' + modelCol + ' Volt:' + voltCol + ' Comp:' + compartmentCol + ' Ah:' + ahCol);
    const mfgClean   = manufacturer.toUpperCase().trim();
    const modelClean = model.toUpperCase().trim().replace(/[^A-Z0-9]/g, '');
    let exactMatches = [];
    let fuzzyMatches = [];
    for (let i = 1; i < data.length; i++) {
      const rowMfg   = String(data[i][mfgCol]).toUpperCase().trim();
      const rowModel = String(data[i][modelCol]).toUpperCase().trim().replace(/[^A-Z0-9]/g, '');
      let compartmentW = 0;
      const compStr = String(data[i][compartmentCol]);
      const compParts = compStr.split('x');
      if (compParts.length >= 2) compartmentW = parseFloat(compParts[1].trim());
const compartmentH = compParts.length >= 3 ? parseFloat(compParts[2].trim()) : null;
      let ah = data[i][ahCol];
      if (!ah || ah === '' || ah === 0) {
        const plateRating = data[i][plateRatingCol];
        if (plateRating) {
          const plateStr = String(plateRating);
          const parts = plateStr.split('-');
          if (parts.length >= 3) {
            const plates = Math.floor(parseInt(parts[2]) / 2);
            const rating = parseInt(parts[1].replace('D', ''));
            ah = plates * rating;
          }
        }
      }
      const rowData = {
  manufacturer: data[i][mfgCol], model: data[i][modelCol],
  voltage: data[i][voltCol], ah: ah,
  compartment_w: compartmentW, compartment_h: compartmentH,
  compartment_full: compStr, row: i + 1
};
      if (rowMfg === mfgClean && rowModel === modelClean) {
        exactMatches.push(rowData);
      } else if (rowMfg === mfgClean && rowModel.includes(modelClean)) {
        fuzzyMatches.push(rowData);
      }
    }
    let filteredExact = exactMatches;
    if (voltage && voltage > 0) {
      filteredExact = exactMatches.filter(m => m.voltage == voltage);
      if (filteredExact.length === 0) filteredExact = exactMatches;
    }
    // Height tiebreaker — if full dimensions were provided by Gemini, use height to resolve
const dims = parseCompartmentInput(params ? (params.compartment || '') : '');
if (dims.height && filteredExact.length > 1) {
  const heightMatch = filteredExact.filter(m =>
    m.compartment_h && Math.abs(m.compartment_h - dims.height) < 0.5
  );
  if (heightMatch.length === 1) {
    Logger.log('Height tiebreaker resolved: ' + JSON.stringify(heightMatch[0]));
    filteredExact = heightMatch;
  }
}
    fuzzyMatches.sort(function(a, b) {
  var aModel = String(a.model).toUpperCase().replace(/[^A-Z0-9]/g, '');
  var bModel = String(b.model).toUpperCase().replace(/[^A-Z0-9]/g, '');
  var aScore = aModel.startsWith(modelClean) ? 0 : 1;
  var bScore = bModel.startsWith(modelClean) ? 0 : 1;
  return aScore - bScore;
});
if (voltage && voltage > 0 && fuzzyMatches.length > 0) {
  var voltFiltered = fuzzyMatches.filter(function(m) { return m.voltage == voltage; });
  if (voltFiltered.length > 0) fuzzyMatches = voltFiltered;
}
    const result = {
      exact_matches: filteredExact, fuzzy_matches: fuzzyMatches,
      total_exact: filteredExact.length, total_fuzzy: fuzzyMatches.length,
      all_voltages: [...new Set(exactMatches.map(m => m.voltage))], status: ''
    };
    if (filteredExact.length === 1) {
      result.status = 'EXACT_SINGLE';
    } else if (filteredExact.length > 1) {
      const uniqueAh = [...new Set(filteredExact.map(m => m.ah))];
      result.status = uniqueAh.length === 1 ? 'EXACT_MULTIPLE_SAME_AH' : 'EXACT_MULTIPLE_DIFFERENT_AH';
    } else if (fuzzyMatches.length > 0) {
      result.status = 'FUZZY_ONLY';
    } else {
      result.status = 'NOT_FOUND';
    }
    Logger.log('Status: ' + result.status + ' | Exact: ' + result.total_exact + ' | Voltages: ' + result.all_voltages);
    if (filteredExact.length > 0) Logger.log('First match: ' + JSON.stringify(filteredExact[0]));
    else if (fuzzyMatches.length > 0) Logger.log('First fuzzy: ' + JSON.stringify(fuzzyMatches[0]));
    return result;
  } catch (e) {
    Logger.log('Error: ' + e.toString());
    return { status: 'ERROR', message: e.toString() };
  }
}

function isBatteryToAvoid(cellValue) {
  if (cellValue === true || cellValue === 1) return true;
  if (typeof cellValue === 'string') return cellValue.trim().toUpperCase() === 'OUI';
  return false;
}

function verifyEbox(compartmentWidth, voltage, compartmentLength, compartmentHeight) {
  const ss             = SpreadsheetApp.getActiveSpreadsheet();
  const eboxSheet      = ss.getSheetByName('Ebox');
  const batteriesSheet = ss.getSheetByName('Batteries');
  const eboxData       = eboxSheet.getDataRange().getValues();
  const batteryData    = batteriesSheet.getDataRange().getValues();
  const eboxHeaders    = eboxData[0];
  const batteryHeaders = batteryData[0];
  const eboxNameCol    = eboxHeaders.indexOf('Ebox');
  const eboxWCol       = eboxHeaders.indexOf('Caisson min W (po)');
  const eboxLCol       = eboxHeaders.indexOf('Caisson min L (po)');
  const eboxHCol       = eboxHeaders.indexOf('Caisson min H (po)');
  const battVoltCol    = batteryHeaders.indexOf('Voltage');
  const battAhCol      = batteryHeaders.indexOf('Ah');
  const battTypeCol    = batteryHeaders.indexOf('Type');
  const battEboxCol    = batteryHeaders.indexOf('Ebox');
  const battAvoidCol   = 9;
  const battChargeCol  = batteryHeaders.indexOf('Charge max A');
  let fittingEboxes = [];
  for (let i = 1; i < eboxData.length; i++) {
    const eboxName = eboxData[i][eboxNameCol];
    const eboxW    = parseFloat(eboxData[i][eboxWCol]);
    const eboxL    = parseFloat(eboxData[i][eboxLCol]);
    const eboxH    = parseFloat(eboxData[i][eboxHCol]);
    const wFits    = eboxW <= compartmentWidth;
    const lFits    = !compartmentLength || isNaN(eboxL) || eboxL <= compartmentLength;
    const hFits    = !compartmentHeight || isNaN(eboxH) || eboxH <= compartmentHeight;
    if (wFits && lFits && hFits) fittingEboxes.push(eboxName);
  }
  Logger.log('Fitting Eboxes: ' + fittingEboxes);
  let compatibleBatteries = [];
  for (let i = 1; i < batteryData.length; i++) {
    const battVolt   = batteryData[i][battVoltCol];
    const battAh     = batteryData[i][battAhCol];
    const battType   = batteryData[i][battTypeCol];
    const battEbox   = batteryData[i][battEboxCol];
    const battAvoid  = batteryData[i][battAvoidCol];
    const battCharge = batteryData[i][battChargeCol];
    if (battVolt == voltage && fittingEboxes.includes(battEbox)) {
      compatibleBatteries.push({
        voltage: battVolt, ah: battAh, type: battType,
        ebox: battEbox, max_charge_a: battCharge,
        avoid: isBatteryToAvoid(battAvoid)
      });
    }
  }
  compatibleBatteries.sort((a, b) => a.ah - b.ah);
  const nonAvoid  = compatibleBatteries.filter(b => !b.avoid);
  const rawList   = nonAvoid.length > 0 ? nonAvoid : compatibleBatteries;
  const seen = new Set();
  const finalList = rawList.filter(b => {
    const key = b.ah + '_' + String(b.type).toUpperCase();
    if (seen.has(key)) return false;
    seen.add(key);
    return true;
  });
  Logger.log('Compatible batteries: ' + JSON.stringify(finalList));
  return {
    compartment_w: compartmentWidth, fitting_eboxes: fittingEboxes,
    compatible_batteries: finalList, total_compatible: finalList.length,
    has_avoid_only: nonAvoid.length === 0 && compatibleBatteries.length > 0
  };
}

// ─────────────────────────────────────────────────────────────────────────────
// CHARGER LOOKUP
// ─────────────────────────────────────────────────────────────────────────────

function getChargersForVoltage(voltage, needsIp54, needsHeating) {
  const ss      = SpreadsheetApp.getActiveSpreadsheet();
  const sheet   = ss.getSheetByName('Chargeurs');
  const data    = sheet.getDataRange().getValues();
  const headers = data[0];
  const voltageColHeader = 'A ' + voltage + 'V';
  const voltageCol = headers.indexOf(voltageColHeader);
  if (voltageCol === -1) {
    Logger.log('No column for voltage: ' + voltage + 'V');
    return [];
  }
  const PN_COL   = 0;
  const TYPE_COL = 1;
  const IP54_COL = headers.indexOf('IP54');
  const HEAT_COL = headers.indexOf('Chauffé');
  const NOTE_COL = headers.indexOf('Notes');
  const chargers = [];
  for (let i = 1; i < data.length; i++) {
    const model = String(data[i][PN_COL] || '').trim();
    if (!model) continue;
    const type = String(data[i][TYPE_COL] || '').trim().toUpperCase();
    if (type !== 'VELOX') continue;
    const notes = String(data[i][NOTE_COL] || '').trim();
    if (notes.toLowerCase().includes('usa')) continue;
    if ((notes.toLowerCase().includes('48v') || notes.toLowerCase().includes('48 v')) && voltage > 48) continue;
    const amps = Number(data[i][voltageCol]);
    if (!amps || amps === 0) continue;
    const ip54   = String(data[i][IP54_COL] || '').trim().toUpperCase() === 'OUI';
    const heated = String(data[i][HEAT_COL] || '').trim().toUpperCase() === 'OUI';
    if (needsIp54    && !ip54)   continue;
    if (needsHeating && !heated) continue;
    if (!needsIp54 && !needsHeating && (ip54 || heated)) continue;
    chargers.push({ amps, model, ip54, heated, notes });
  }
  chargers.sort((a, b) => a.amps - b.amps);
  Logger.log('Chargers ' + voltage + 'V: ' + JSON.stringify(chargers));
  return chargers;
}

// ─────────────────────────────────────────────────────────────────────────────
// MINIMUM WORK THRESHOLD
// ─────────────────────────────────────────────────────────────────────────────

function meetsMinWorkThreshold(workMinutes, shifts, batteryType, shiftDuration) {
  const dur = shiftDuration || 8;
  const isVeloxPlus = String(batteryType || '').toUpperCase().includes('VELOXPLUS') ||
                      String(batteryType || '').toUpperCase().includes('VELOX PLUS');

  if (dur === 12) {
    if (shifts === 1) return workMinutes >= 510;  // 8h30
    if (shifts === 2) {
      if (isVeloxPlus) return workMinutes >= 908;
      return workMinutes >= 630;
    }
  }

  if (shifts === 3) return workMinutes >= 900;    // 5h00 per shift × 3 = 15h00 — same for Flex and Plus
  if (shifts === 1) return workMinutes >= 360;    // 6h00
  if (shifts === 2) return workMinutes >= 660;    // 5h30 per shift × 2 = 11h00
  return workMinutes >= 1080;
}

// ─────────────────────────────────────────────────────────────────────────────
// SIMULATION PLAN SELECTION
// ─────────────────────────────────────────────────────────────────────────────

function selectSimulations(dekaResult, eboxResult, params) {
  const flags = [];
  const plans = [];

  if (!eboxResult.compatible_batteries || eboxResult.compatible_batteries.length === 0) {
    return { plans: [], flags: ['AUCUNE_BATTERIE_COMPATIBLE'], stopReason: 'Aucune batterie compatible.' };
  }
  if (eboxResult.has_avoid_only) flags.push('BATTERIES_A_EVITER_SEULEMENT');

  const voltage    = params.voltage || (dekaResult.exact_matches[0] && dekaResult.exact_matches[0].voltage) || 0;
  const useType    = params.use_type || 'average';
  const shifts     = params.shifts || 2;
  const shiftDur   = params.shift_duration || 8;
  const bestMatch  = dekaResult.exact_matches[0] || dekaResult.fuzzy_matches[0] || null;
  const leadAcidAh = bestMatch ? bestMatch.ah : 0;
  const compartW   = bestMatch ? bestMatch.compartment_w : 0;
  const divisor    = useType === 'heavy' ? 8 : useType === 'moderate' ? 9 : 10;
  const truckLoad  = leadAcidAh > 0 ? Math.round(leadAcidAh * 0.80 / divisor) : 0;
  const truckEff   = useType === 'light' ? 50 : 80;

  if (!leadAcidAh) flags.push('AH_REF_MANQUANT');
  if (!params.shifts) flags.push('SHIFTS_ASSUMES_2_PAR_DEFAUT');

  const allBatteries = eboxResult.compatible_batteries.slice().sort((a, b) => a.ah - b.ah);
  const chargers     = getChargersForVoltage(voltage, params.ip54_charger, params.heating_option);
  if (chargers.length === 0) flags.push('AUCUN_CHARGEUR_POUR_' + voltage + 'V');

  const isVeloxPlusType = function(b) {
    return String(b.type || '').toUpperCase().includes('VELOXPLUS') ||
           String(b.type || '').toUpperCase().includes('VELOX PLUS');
  };

  const findCharger = function(amps) {
    return chargers.find(function(c) { return c.amps === amps; }) || null;
  };

// ── Dealer specified exact combo — add as first plan then continue ────────
  if (params.battery_ah > 0 && params.charger_amps > 0) {
    flags.push('COMBO_SPECIFIE_PAR_DEALER');

    const specBattType = params.battery_ah >= 550 ? 'VeloxPlus' : 'VeloxFlex';
    const specCharger  = chargers.find(function(c) { return c.amps === params.charger_amps; });
    const specChargerModel = specCharger ? specCharger.model : '';

    // Add dealer combo as first plan
    plans.push({
      battery_ah:     params.battery_ah,
      charger_amps:   params.charger_amps,
      charger_model:  specChargerModel,
      truck_load:     truckLoad,
      truck_efficiency: truckEff,
      lead_acid_ah:   leadAcidAh,
      compartment_w:  compartW,
      shifts, shift_duration: shiftDur,
      label:          'Sim 1 — ' + params.battery_ah + 'Ah + ' + params.charger_amps + 'A (Dealer Specified)',
      conditional:    false,
      battery_type:   specBattType,
      ebox:           (eboxResult.compatible_batteries.find(function(b) { return b.ah === params.battery_ah; }) || {}).ebox || '',
      simIndex:       1
    });

    // Zero out so standard pipeline builds remaining plans normally
    params.battery_ah   = 0;
    params.charger_amps = 0;

    // Fall through to standard pipeline below
  }

  // ── 24h operations (3x8h or 2x12h) ───────────────────────────────────────
  const is24hOperation = (shifts === 3 && shiftDur === 8) || (shifts === 2 && shiftDur === 12);

  if (is24hOperation) {
    const batteries24 = allBatteries.filter(function(b) { return b.ah >= 450; });
    if (batteries24.length === 0) {
      return { plans: [], flags: ['AUCUNE_BATTERIE_24H'], stopReason: 'Aucune batterie >=450Ah pour application 24h.' };
    }

    const flexBatteries = batteries24.filter(function(b) { return !isVeloxPlusType(b); });
    const plusBatteries = batteries24.filter(function(b) { return isVeloxPlusType(b); });
    let simIdx = 1;

    // ── 24h Flex: 1 sim only, cap 275A ───────────────────────────────────
    // If 18KVA charger (275A for 48V, 360A for 36V) exists, use it directly
    // as the Flex charger — no injection needed, no second sim produced.
    const flex450 = flexBatteries.find(function(b) { return b.ah === 450; }) || flexBatteries[0];
    if (flex450) {
      const flexStartAmps = voltage === 80 ? 215 : 250;
      const flexCapAmps   = voltage === 80 ? 215 : 275;
      const kva18Amps     = voltage === 36 ? 360 : voltage === 48 ? 275 : voltage === 24 ? 400 : 0;
      const kva18Charger  = kva18Amps > 0 ? findCharger(kva18Amps) : null;
      const flexCh = kva18Charger ||
                     findCharger(flexStartAmps) ||
                     chargers.find(function(c) { return c.amps >= flexStartAmps && c.amps <= flexCapAmps; }) ||
                     chargers[0];
      if (flexCh) {
        plans.push({
          battery_ah: flex450.ah, charger_amps: flexCh.amps, charger_model: flexCh.model,
          truck_load: truckLoad, truck_efficiency: truckEff, lead_acid_ah: leadAcidAh,
          compartment_w: compartW, shifts, shift_duration: shiftDur,
          label: 'Sim ' + simIdx + ' — ' + flex450.ah + 'Ah + ' + flexCh.amps + 'A (VeloxFlex)',
          battery_type: flex450.type || 'VeloxFlex', ebox: flex450.ebox || '',
          conditional: false, simIndex: simIdx,
          check18KVAThreshold: false  // charger already selected optimally above
          // TODO: add 320A threshold logic here when rule is defined
        });
        simIdx++;
      }
    }

    // VeloxPlus sequences
    // ── 24h Plus: 1 sim only, highest charger available ──────────────────
    const plus550 = plusBatteries.find(function(b) { return b.ah === 550; }) || plusBatteries[0];
    const plus700 = plusBatteries.find(function(b) { return b.ah === 700; });
    const plusTopSeq = voltage === 80 ? [400, 335, 270]
                     : voltage === 72 ? [475, 375, 310]
                     : [400, 360, 320];

    if (plus550) {
      const plusCh = plusTopSeq.reduce(function(found, amps) {
        return found || findCharger(amps);
      }, null);
      if (plusCh) {
        plans.push({
          battery_ah: plus550.ah, charger_amps: plusCh.amps, charger_model: plusCh.model,
          truck_load: truckLoad, truck_efficiency: truckEff, lead_acid_ah: leadAcidAh,
          compartment_w: compartW, shifts, shift_duration: shiftDur,
          label: 'Sim ' + simIdx + ' — ' + plus550.ah + 'Ah + ' + plusCh.amps + 'A (VeloxPlus)',
          battery_type: plus550.type || 'VeloxPlus', ebox: plus550.ebox || '',
          conditional: false, simIndex: simIdx
        });
        simIdx++;
      }
    } else if (plus700) {
      const plusCh = plusTopSeq.reduce(function(found, amps) {
        return found || findCharger(amps);
      }, null);
      if (plusCh) {
        plans.push({
          battery_ah: plus700.ah, charger_amps: plusCh.amps, charger_model: plusCh.model,
          truck_load: truckLoad, truck_efficiency: truckEff, lead_acid_ah: leadAcidAh,
          compartment_w: compartW, shifts, shift_duration: shiftDur,
          label: 'Sim ' + simIdx + ' — ' + plus700.ah + 'Ah + ' + plusCh.amps + 'A (VeloxPlus)',
          battery_type: plus700.type || 'VeloxPlus', ebox: plus700.ebox || '',
          conditional: false, simIndex: simIdx
        });
        simIdx++;
      }
    }

    // Deduplicate dealer combo from 24h plans too
    const dealerPlan24 = plans.find(function(p) {
      return p.label && p.label.includes('Dealer Specified');
    });
    const finalPlans24 = dealerPlan24
      ? plans.filter(function(p) {
          if (p.label && p.label.includes('Dealer Specified')) return true;
          return !(p.battery_ah === dealerPlan24.battery_ah &&
                   p.charger_amps === dealerPlan24.charger_amps);
        })
      : plans;
    return { plans: finalPlans24, flags, stopReason: null };
  }

  // ── STANDARD PATH (1x8h, 2x8h, 1x12h) ───────────────────────────────────
  const include280 = leadAcidAh <= 510 || allBatteries.every(function(b) { return b.ah === 280; });
  if (!include280) flags.push('280AH_EXCLU_REF_LA_' + leadAcidAh + 'AH');
  const batteries = include280 ? allBatteries : allBatteries.filter(function(b) { return b.ah > 280; });
  const effectiveBatteries = batteries.length > 0 ? batteries : allBatteries;

  // Ebox-constrained: only one battery fits — vary charger
  if (effectiveBatteries.length === 1 && chargers.length > 1) {
    flags.push('EBOX_CONTRAINT_VARIATION_CHARGEUR');
    const maxSafeAmps = Math.floor(effectiveBatteries[0].ah * 0.80);
    const minFlexAmps  = voltage === 80 ? 215 : voltage === 72 ? 215 : 250;
    const safeChargers = chargers.filter(function(c) { return c.amps >= minFlexAmps && c.amps <= maxSafeAmps; });
    const chargersToUse = safeChargers.length > 0 ? safeChargers : [chargers[0]];
    chargersToUse.forEach(function(charger, idx) {
      plans.push({
        battery_ah: effectiveBatteries[0].ah, charger_amps: charger.amps, charger_model: charger.model,
        truck_load: truckLoad, truck_efficiency: truckEff, lead_acid_ah: leadAcidAh,
        compartment_w: compartW, shifts, shift_duration: shiftDur,
        label: 'Sim ' + (idx + 1) + ' — ' + effectiveBatteries[0].ah + 'Ah + ' + charger.amps + 'A',
        battery_type: effectiveBatteries[0].type || 'VeloxFlex',
        ebox: effectiveBatteries[0].ebox || '', conditional: false
      });
    });
    return { plans, flags, stopReason: null };
  }

  let simIdx = 1;

  // ── 280Ah: lowest available charger first ─────────────────────────────────
  const bat280 = include280 ? effectiveBatteries.find(function(b) { return b.ah === 280; }) : null;
  if (bat280) {
    const maxSafe280 = Math.floor(bat280.ah * 0.80); // 224A for 280Ah
    const safe280Chargers = chargers.filter(function(c) { return c.amps <= maxSafe280; });
    const ch170 = findCharger(170) || (safe280Chargers.length > 0 ? safe280Chargers[0] : null);
    const ch215 = safe280Chargers.find(function(c) { return c.amps === 215; }) ||
                  (safe280Chargers.length > 1 ? safe280Chargers[1] : null);
    if (ch170) {
      plans.push({
        battery_ah: bat280.ah, charger_amps: ch170.amps, charger_model: ch170.model,
        truck_load: truckLoad, truck_efficiency: truckEff, lead_acid_ah: leadAcidAh,
        compartment_w: compartW, shifts, shift_duration: shiftDur,
        label: 'Sim ' + simIdx + ' — ' + bat280.ah + 'Ah + ' + ch170.amps + 'A (VeloxFlex)',
        battery_type: bat280.type || 'VeloxFlex', ebox: bat280.ebox || '',
        conditional: false, simIndex: simIdx
      });
      simIdx++;
    }
    if (ch215 && ch215.amps !== (ch170 ? ch170.amps : 0)) {
      plans.push({
        battery_ah: bat280.ah, charger_amps: ch215.amps, charger_model: ch215.model,
        truck_load: truckLoad, truck_efficiency: truckEff, lead_acid_ah: leadAcidAh,
        compartment_w: compartW, shifts, shift_duration: shiftDur,
        label: 'Sim ' + simIdx + ' — ' + bat280.ah + 'Ah + ' + ch215.amps + 'A (VeloxFlex)',
        battery_type: bat280.type || 'VeloxFlex', ebox: bat280.ebox || '',
        conditional: true, conditionAfterSim: simIdx - 1, simIndex: simIdx
      });
      simIdx++;
    }
  }

  // ── VeloxFlex 450Ah: 1 sim only, cap 275A ────────────────────────────────
  // If 18KVA charger exists for this voltage, use it directly (replaces 250A — no injection).
  // TODO: add 320A threshold logic here when rule is defined
  const bat450 = effectiveBatteries.find(function(b) { return b.ah === 450; }) ||
                 effectiveBatteries.find(function(b) { return !isVeloxPlusType(b) && b.ah > 280; });
  if (bat450) {
    const flexStartAmps = voltage === 80 ? 215 : 250;
    const flexCapAmps   = voltage === 80 ? 215 : 275;
    const kva18Amps     = voltage === 36 ? 360 : voltage === 48 ? 275 : voltage === 24 ? 400 : 0;
    const kva18Charger  = kva18Amps > 0 ? findCharger(kva18Amps) : null;
    const ch1 = kva18Charger ||
                findCharger(flexStartAmps) ||
                chargers.find(function(c) { return c.amps >= flexStartAmps && c.amps <= flexCapAmps; }) ||
                chargers[0];
    if (ch1) {
      plans.push({
        battery_ah: bat450.ah, charger_amps: ch1.amps, charger_model: ch1.model,
        truck_load: truckLoad, truck_efficiency: truckEff, lead_acid_ah: leadAcidAh,
        compartment_w: compartW, shifts, shift_duration: shiftDur,
        label: 'Sim ' + simIdx + ' — ' + bat450.ah + 'Ah + ' + ch1.amps + 'A (VeloxFlex)',
        battery_type: bat450.type || 'VeloxFlex', ebox: bat450.ebox || '',
        conditional: false, simIndex: simIdx,
        check18KVAThreshold: false  // charger already selected optimally above
      });
      simIdx++;
    }
  }

 // ── VeloxPlus: 1 sim only, highest charger available ─────────────────────
  // For 2×8h: gating in runPass still applies (skips if Flex passed ≤210% utilization)
  const plusBat = effectiveBatteries.find(function(b) { return isVeloxPlusType(b); });
  if (plusBat) {
    const plusTopSeq = voltage === 80 ? [400, 335, 270]
                     : voltage === 72 ? [475, 375, 310]
                     : [400, 360, 320];
    const plusCh = plusTopSeq.reduce(function(found, amps) {
      return found || findCharger(amps);
    }, null);
    if (plusCh) {
      plans.push({
        battery_ah: plusBat.ah, charger_amps: plusCh.amps, charger_model: plusCh.model,
        truck_load: truckLoad, truck_efficiency: truckEff, lead_acid_ah: leadAcidAh,
        compartment_w: compartW, shifts, shift_duration: shiftDur,
        label: 'Sim ' + simIdx + ' — ' + plusBat.ah + 'Ah + ' + plusCh.amps + 'A (VeloxPlus)',
        battery_type: plusBat.type || 'VeloxPlus', ebox: plusBat.ebox || '',
        isVeloxPlusPlan: true,
        isFirstPlus: true,
        conditional: false,
        simIndex: simIdx
      });
      simIdx++;
    }
  }

  // Remove any standard plan that duplicates the dealer-specified combo
  const dealerPlan = plans.find(function(p) {
    return p.label && p.label.includes('Dealer Specified');
  });
  const finalPlans = dealerPlan
    ? plans.filter(function(p, idx) {
        if (p.label && p.label.includes('Dealer Specified')) return true;
        return !(p.battery_ah === dealerPlan.battery_ah &&
                 p.charger_amps === dealerPlan.charger_amps);
      })
    : plans;

  return { plans: finalPlans, flags, stopReason: null, chargers: chargers };
}

function timeToMinutes(timeStr) {
  const str   = String(timeStr).trim().toUpperCase();
  const match = str.match(/^(\d+):(\d+)\s*(AM|PM)$/);
  if (!match) return 0;
  let hours    = parseInt(match[1]);
  const mins   = parseInt(match[2]);
  const period = match[3];
  if (period === 'PM' && hours !== 12) hours += 12;
  if (period === 'AM' && hours === 12) hours = 0;
  return hours * 60 + mins;
}

function minutesToHours(minutes) {
  const h = Math.floor(minutes / 60);
  const m = minutes % 60;
  return h + 'h' + (m < 10 ? '0' + m : m);
}

function calculateWorkHours(chargingEvents, shifts, truckEfficiency, shiftDuration, shiftStartHour) {
  const dur = shiftDuration || 8;
  const shiftWindows = [];

 const startH = (shiftStartHour !== undefined && shiftStartHour !== null) ? shiftStartHour : 7;
  const startMin = startH * 60;

  if (shifts === 1 && dur === 8) {
    shiftWindows.push({ start: startMin, end: startMin + 8*60, name: 'S1' });
  } else if (shifts === 1 && dur === 12) {
    shiftWindows.push({ start: startMin, end: startMin + 12*60, name: 'S1' });
  } else if (shifts === 2 && dur === 8) {
    shiftWindows.push({ start: startMin,        end: startMin + 8*60,  name: 'S1' });
    shiftWindows.push({ start: startMin + 8*60, end: startMin + 16*60, name: 'S2' });
  } else if (shifts === 2 && dur === 12) {
    shiftWindows.push({ start: 0,     end: 12*60, name: 'S1' });
    shiftWindows.push({ start: 12*60, end: 24*60, name: 'S2' });
  } else {
    shiftWindows.push({ start: 0,     end: 8*60,  name: 'S1' });
    shiftWindows.push({ start: 8*60,  end: 16*60, name: 'S2' });
    shiftWindows.push({ start: 16*60, end: 24*60, name: 'S3' });
  }

  const efficiency     = (truckEfficiency || 80) / 100;
  let totalWorkMinutes = 0;
  const perShift       = [];
  shiftWindows.forEach(function(window) {
    const shiftLength = window.end - window.start;
    let chargeInShift = 0;
    chargingEvents.forEach(function(event) {
      const eStart = timeToMinutes(event.start_time);
      let eEnd     = timeToMinutes(event.end_time);
      if (String(event.end_time).toUpperCase().includes('11:59')) eEnd = 1440;
      const overlapStart = Math.max(eStart, window.start);
      const overlapEnd   = Math.min(eEnd,   window.end);
      if (overlapEnd > overlapStart) chargeInShift += overlapEnd - overlapStart;
    });
    const available   = shiftLength - chargeInShift;
    const workMinutes = Math.round(available * efficiency);
    totalWorkMinutes += workMinutes;
    perShift.push({ name: window.name, workMinutes: workMinutes });
    Logger.log(window.name + ': ' + available + 'min avail x ' + (efficiency*100) + '% = ' + workMinutes + 'min work');
  });
  return { totalMinutes: totalWorkMinutes, formatted: minutesToHours(totalWorkMinutes), perShift: perShift };
}

function analyzeSOC(socArray, shifts, shiftDuration) {
  const dur = shiftDuration || 8;
  let minSOC = 100;
  for (let i = 0; i < socArray.length; i++) {
    const val = socArray[i];
    if (val > 0 && val < minSOC) minSOC = val;
  }

  const result = { min_soc: minSOC };

  if (shifts === 3 && dur === 8) {
    let peakS1 = 0;
    for (let i = 0; i < 480; i++) { if (socArray[i] > peakS1) peakS1 = socArray[i]; }
    let peakS2 = 0;
    for (let i = 480; i < 960; i++) { if (socArray[i] > peakS2) peakS2 = socArray[i]; }
    let peakS3 = 0;
    for (let i = 960; i < 1440; i++) { if (socArray[i] > peakS3) peakS3 = socArray[i]; }
    result.peak_soc_s1   = peakS1 >= 99.0 ? 100 : peakS1;
    result.peak_soc_s2   = peakS2 >= 99.0 ? 100 : peakS2;
    result.peak_soc_s3   = peakS3 >= 99.0 ? 100 : peakS3;
    result.boundary_pass = (peakS1 >= 99.0 && peakS2 >= 99.0 && peakS3 >= 99.0);
    Logger.log('  Peak S1: ' + Math.round(result.peak_soc_s1) + '% | S2: ' + Math.round(result.peak_soc_s2) + '% | S3: ' + Math.round(result.peak_soc_s3) + '% | Pass: ' + result.boundary_pass);
  } else if (shifts === 2 && dur === 12) {
    let peakS1 = 0;
    for (let i = 0; i < 720; i++) { if (socArray[i] > peakS1) peakS1 = socArray[i]; }
    let peakS2 = 0;
    for (let i = 720; i < 1440; i++) { if (socArray[i] > peakS2) peakS2 = socArray[i]; }
    result.peak_soc_s1   = peakS1 >= 99.0 ? 100 : peakS1;
    result.peak_soc_s2   = peakS2 >= 99.0 ? 100 : peakS2;
    result.boundary_pass = (peakS1 >= 99.0 && peakS2 >= 99.0);
    Logger.log('  Peak S1: ' + Math.round(result.peak_soc_s1) + '% | S2: ' + Math.round(result.peak_soc_s2) + '% | Pass: ' + result.boundary_pass);
  }

  Logger.log('  Min SOC: ' + Math.round(minSOC) + '%');
  return result;
}

// EFFICIENCY SEARCH
// ─────────────────────────────────────────────────────────────────────────────

function runSingleSim(plan, chargingEvents, truckEfficiency) {
  const simParams = {
    battery_ah:       plan.battery_ah,
    truck_load:       plan.truck_load,
    truck_efficiency: truckEfficiency,
    charger_amps:     plan.charger_amps,
    shift1_label:     plan.shift1_label || '',
    shift2_label:     plan.shift2_label || '',
    shift3_label:     plan.shift3_label || '',
    charging_events:  chargingEvents
  };
  writeSimulationToSheet(simParams);
  const results   = readSimulationResults();
  const socArray  = readAllSOC();
  const shiftDur  = plan.shift_duration || 8;
  const socInfo   = analyzeSOC(socArray, plan.shifts || 2, shiftDur);

  results.min_soc     = socInfo.min_soc;
  results.peak_soc_s1 = socInfo.peak_soc_s1;
  results.peak_soc_s2 = socInfo.peak_soc_s2;
  results.peak_soc_s3 = socInfo.peak_soc_s3;

  const is24h = (plan.shifts === 3 && shiftDur === 8) || (plan.shifts === 2 && shiftDur === 12);
  if (is24h && socInfo.boundary_pass === false) results.status = 'FAIL';
  if (results.min_soc < 20) results.status = 'FAIL';

  return results;
}

function findMaxEfficiency(plan, chargingEvents, veloxFlexMaxEff, targetEfficiency) {
  const isVeloxPlus = String(plan.battery_type || '').toUpperCase().includes('VELOXPLUS') ||
                      String(plan.battery_type || '').toUpperCase().includes('VELOX PLUS');
  const startEff   = isVeloxPlus ? 70 : 45;
  const shiftDur   = plan.shift_duration || 8;
  const is24h      = (plan.shifts === 3 && shiftDur === 8) || (plan.shifts === 2 && shiftDur === 12);
  const coarseStep = is24h ? 5 : 10;
  const target     = (targetEfficiency !== undefined && targetEfficiency !== null)
    ? targetEfficiency : 100;

  Logger.log('  Search | ' + (isVeloxPlus ? 'VeloxPlus' : 'VeloxFlex') + ' | Start: ' + startEff + '% | Step: ' + coarseStep + '%');

  let lastPassEff     = 0;
  let lastPassResults = null;
  let flag            = null;
  let coarseFailAt    = -1;

  for (let eff = startEff; eff <= target; eff += coarseStep) {
    const results = runSingleSim(plan, chargingEvents, eff);
    Logger.log('  Coarse ' + eff + '% -> ' + results.status);
    if (results.status === 'PASS') {
      lastPassEff = eff; lastPassResults = results;
      if (eff >= target) break;
    } else {
      coarseFailAt = eff; break;
    }
  }

  if (coarseFailAt === -1 && lastPassEff < target) {
    const rTarget = runSingleSim(plan, chargingEvents, target);
    Logger.log('  Coarse 100% -> ' + rTarget.status);
    if (rTarget.status === 'PASS') { lastPassEff = target; lastPassResults = rTarget; }
    else coarseFailAt = target;
  }

  if (coarseFailAt > startEff) {
    const fineStart = coarseFailAt - (coarseStep - 1);
    for (let eff = fineStart; eff < coarseFailAt; eff++) {
      const results = runSingleSim(plan, chargingEvents, eff);
      Logger.log('  Fine ' + eff + '% -> ' + results.status);
      if (results.status === 'PASS') { lastPassEff = eff; lastPassResults = results; }
      else break;
    }
  } else if (coarseFailAt === startEff || lastPassEff === 0) {
    lastPassEff     = startEff;
    lastPassResults = runSingleSim(plan, chargingEvents, startEff);
    flag            = 'COMBO_FAIBLE_EFF_MIN_' + startEff + 'PCT';
    Logger.log('  Failed at start — reporting at ' + startEff + '%');
  }
  if (isVeloxPlus && veloxFlexMaxEff > 0) {
    const required = veloxFlexMaxEff + 15;
    if (lastPassEff < required) {
      flag = (flag ? flag + ' | ' : '') + 'PAS_D_AVANTAGE_SUFFISANT — VeloxFlex: ' + veloxFlexMaxEff + '% | VeloxPlus: ' + lastPassEff + '% | Requis: +15pp=' + required + '%';
    }
  }

  if (plan.dualPortCheck && lastPassEff < 75) {
    flag = (flag ? flag + ' | ' : '') + 'DOUBLE_PORT_SUGGERE — 2x270A=540A effectif recommande';
    Logger.log('  Dual port suggested');
  }

  if (lastPassResults && lastPassResults.battery_used_pct >= 220) {
    flag = (flag ? flag + ' | ' : '') + 'MARGE_FAIBLE — ' + Math.round(lastPassResults.battery_used_pct) + '%';
  }

  Logger.log('  Max efficiency: ' + lastPassEff + '%');
  return { maxEfficiency: lastPassEff, finalResults: lastPassResults, flag: flag };
}

function capturePdf(params, plan, simNumber, scheduleTag) {
  const ss      = SpreadsheetApp.getActiveSpreadsheet();
  const sheetId = ss.getSheetByName('Data Entry').getSheetId();
  const ssId    = ss.getId();
  const exportUrl = `https://docs.google.com/spreadsheets/d/${ssId}/export?format=pdf&gid=${sheetId}&single=true&portrait=false&fitw=true&gridlines=false&printtitle=false&sheetnames=false&pagesize=A4&scale=4&top_margin=0.5&bottom_margin=0.5&left_margin=0.5&right_margin=0.5`;
  const token = ScriptApp.getOAuthToken();
  const blob  = UrlFetchApp.fetch(exportUrl, { headers: { Authorization: 'Bearer ' + token } })
    .getBlob().setContentType('application/pdf');
  const schedPart = scheduleTag ? '_' + scheduleTag : '';
const filePrefix = (params.language || 'fr').toLowerCase() === 'en' ? 'Scope_ ' : 'Portée_ ';
  const fileName  = (filePrefix + (params.dealer || 'Dealer') + ' - ' + (params.customer || 'Client') + ' - ' +

                    .replace(/[^a-zA-ZÀ-ÿ0-9_\-. ]/g, '_');
  blob.setName(fileName);
  Logger.log('PDF: ' + fileName);
  return blob;
}

function getDefaultChargeMinutes(shifts, shiftDuration) {
  const dur = shiftDuration || 8;
  if (dur === 12) return 90;
  return 60;
}

// ─────────────────────────────────────────────────────────────────────────────
// MASTER PIPELINE
// ─────────────────────────────────────────────────────────────────────────────

function runSimulationPipeline(params) {
  Logger.log('Pipeline: ' + params.forklift_manufacturer + ' ' + params.forklift_model);

const pipelineFlags    = [];
  const allSimResults    = [];
const allPdfs          = [];
const allSimScreenshots = [];

  // ── Basic validation ──────────────────────────────────────────────────────
  if (!params.forklift_manufacturer || !params.forklift_model) {
    sendFlagEmail(params, 'MODELE_MANQUANT', 'Fabricant ou modèle manquant.', params.thread);
    return;
  }
  if (!params.shifts || params.shifts === 0) {
    params.shifts = 2;
    pipelineFlags.push('SHIFTS_ASSUMES_2_PAR_DEFAUT');
  }
  if (!params.shift_duration) params.shift_duration = 8;
  if (!params.use_type) params.use_type = 'average';

  // ── Sender ────────────────────────────────────────────────────────────────
  const senderLabel = [params.sender_name || '', params.sender_email || '']
                        .filter(Boolean).join(' ');
  params.sender = senderLabel;

  // ── SIM_DB lookup — always attempt for known clients ─────────────────────
  const refModel  = params.referenced_model || params.forklift_model || '';
  const simDbHit = params._skipSimDb ? null : lookupPreviousSimulation(params.dealer, refModel, params.customer);

  if (simDbHit) {
    const prev      = simDbHit.data;
    const matchType = simDbHit.match;
    Logger.log('SIM_DB ' + matchType + ' match — filling params');
    pipelineFlags.push('PARAMS_DEPUIS_SIM_DB_' + matchType + ' — ' + prev.fabricant + ' ' + prev.model + ' | Client: ' + (params.customer || ''));

    // Exact match: fill all missing params including voltage and battery
    if (matchType === 'EXACT') {
      if (!params.voltage || params.voltage === 0)           params.voltage        = prev.voltage;
      if (!params.shifts  || params.shifts === 0)            params.shifts         = prev.shifts;
      if (!params.shift_duration || params.shift_duration === 8) params.shift_duration = prev.shift_duration;
      if (!params.use_type || params.use_type === 'average') params.use_type       = prev.use_type;
      if (!params.compartment || params.compartment === '')  params.compartment    = String(prev.compartment_w);
      if (!params.forklift_manufacturer || params.forklift_manufacturer === '') params.forklift_manufacturer = prev.fabricant;
      // Never carry over battery_ah or charger_amps from SIM_DB — those are model-specific
      // and would incorrectly trigger the COMBO_SPECIFIE_PAR_DEALER shortcut
      params.battery_ah    = 0;
      params.charger_amps  = 0;
      params._simDbRowUsed = simDbHit.rowIndex; // track row for in-place update
      // Never let SIM_DB override explicitly stated charge constraints
      // stated_charge_minutes_per_shift comes from the email and must be preserved
      // (already preserved since we don't touch it above — this comment is a reminder)
      params._simDbRowUsed = simDbHit.rowIndex;
    }

    // Client match: only fill operation params — NOT voltage/battery (different forklift)
  if (matchType === 'CLIENT') {
      if (!params.shifts  || params.shifts === 0)            params.shifts         = prev.shifts;
      if (!params.shift_duration || params.shift_duration === 8) params.shift_duration = prev.shift_duration;
      if (!params.use_type || params.use_type === 'average') params.use_type       = prev.use_type;
      // Never carry over battery/charger from CLIENT match — different forklift
      params.battery_ah   = 0;
      params.charger_amps = 0;
    }
  }

  // ── Stated charge floor check ─────────────────────────────────────────────
  const stated = params.stated_charge_minutes_per_shift || 0;
  const floorMins = params.shift_duration === 12 ? 60 : 45;
  if (stated > 0 && stated < floorMins) {
    sendFlagEmail(params,
      'CHARGE_INSUFFISANTE_MIN_' + floorMins + 'MIN',
      'Client a indiqué ' + stated + ' min/shift. Minimum requis: ' + floorMins + ' min pour shifts de ' + params.shift_duration + 'h.',
      params.thread);
    return;
  }

  // Normalize French decimal separators
if (params.compartment) {
  params.compartment = params.compartment.replace(/(\d),(\d)/g, '$1.$2');
}

// ── Deka lookup ───────────────────────────────────────────────────────────
const dekaResult = lookupDekaPull(params.forklift_manufacturer, params.forklift_model, params.voltage || 0, params);
if (dekaResult.status === 'NOT_FOUND') {
  sendFlagEmail(params, 'MODELE_INTROUVABLE', 'Modèle introuvable dans Deka Pull.', params.thread);
  return;
}

// ── Auto-correct swapped L/W dimensions ──────────────────────────────────
if (params.compartment) {
  var compParts = params.compartment.split(/[xX×]/).map(function(p) { return p.trim(); });
  if (compParts.length === 3) {
    var extractedL = parseFloat(compParts[0]);
    var extractedW = parseFloat(compParts[1]);
    var extractedH = parseFloat(compParts[2]);
    var currentWidthMatches = dekaResult.exact_matches.filter(function(m) {
      return Math.abs(m.compartment_w - extractedW) < 0.5;
    });
    if (currentWidthMatches.length === 0) {
      var swappedWidthMatches = dekaResult.exact_matches.filter(function(m) {
        return Math.abs(m.compartment_w - extractedL) < 0.75;
      });
      if (swappedWidthMatches.length > 0) {
        var dekaEntry = swappedWidthMatches[0];
        var correctedComp = dekaEntry.compartment_full.trim();
        Logger.log('Auto-corrected swapped L/W — Original: ' + params.compartment + ' → Corrected: ' + correctedComp);
        pipelineFlags.push('DIMENSIONS_LW_INVERSEES_AUTO_CORRIGEES — Original: ' + params.compartment + ' → Corrigé: ' + correctedComp);
        params.compartment = correctedComp;
      }
    }
  }
}

// ── EXACT_MULTIPLE_DIFFERENT_AH resolution ────────────────────────────────
if (dekaResult.status === 'EXACT_MULTIPLE_DIFFERENT_AH') {
  var emailComp = (params.compartment || '').trim().replace(/\s+/g, '');
  var voltageMatches = dekaResult.exact_matches.filter(function(m) {
    return m.voltage === params.voltage;
  });

  // Try full compartment string match first
  var fullMatch = voltageMatches.filter(function(m) {
    return (m.compartment_full || '').trim().replace(/\s+/g, '') === emailComp;
  });

  // Fall back to width-only match
  var compW = emailComp ? parseFloat(params.compartment.split('x')[1]) : 0;
  var widthMatch = compW > 0 ? voltageMatches.filter(function(m) {
    return Math.abs(m.compartment_w - compW) < 0.5;
  }) : [];

  var resolved = fullMatch.length === 1 ? fullMatch
               : widthMatch.length === 1 ? widthMatch
               : [];

  if (resolved.length === 1) {
    Logger.log('✅ Resolved by compartment: ' + resolved[0].ah + 'Ah');
    dekaResult.exact_matches = resolved;
    dekaResult.status = 'EXACT_SINGLE';
  } else {
    sendFlagEmail(params, 'CONFIRMATION_REQUISE',
      'Plusieurs Ah: ' + (voltageMatches.length ? voltageMatches : dekaResult.exact_matches)
        .map(function(m){ return m.ah + 'Ah'; }).join(', ') + '. Compartiment requis.',
      params.thread, dekaResult.exact_matches);
    return;
  }
}
if (dekaResult.status === 'FUZZY_ONLY') pipelineFlags.push('CORRESPONDANCE_APPROXIMATIVE_DEKA');

const bestMatch    = dekaResult.exact_matches[0] || dekaResult.fuzzy_matches[0];
const compartmentW = bestMatch.compartment_w;
const leadAcidAh   = bestMatch.ah;

if (!params.voltage || params.voltage === 0) {
  params.voltage = bestMatch.voltage;
  pipelineFlags.push('VOLTAGE_ASSUME_DEPUIS_DEKA');
} else if (params.voltage !== bestMatch.voltage && bestMatch.voltage > 0) {
  pipelineFlags.push('VOLTAGE_CORRIGE_DEKA — Email: ' + params.voltage + 'V | Deka: ' + bestMatch.voltage + 'V');
  params.voltage = bestMatch.voltage;
}
  Logger.log('Deka: ' + leadAcidAh + 'Ah | ' + compartmentW + '" | ' + params.voltage + 'V');

  // ── Ebox + battery selection ──────────────────────────────────────────────
  const compartmentDims = bestMatch.compartment_full
    ? bestMatch.compartment_full.split('x').map(function(p) { return parseFloat(p.trim()); })
    : [];
  const compartmentL = compartmentDims[0] || 0;
  const compartmentH = compartmentDims[2] || 0;
  const eboxResult = verifyEbox(compartmentW, params.voltage, compartmentL, compartmentH);
  if (eboxResult.total_compatible === 0) {
    const eboxFailDetails = eboxResult.fitting_eboxes.length > 0
      ? 'Eboxes physiquement compatibles: ' + eboxResult.fitting_eboxes.join(', ') + ' — mais aucune batterie assignée à ces Eboxes.'
      : 'Aucun Ebox ne rentre dans ce compartiment (L: ' + compartmentL + '" x W: ' + compartmentW + '" x H: ' + compartmentH + '"). EBOX01 requiert min 38.44" L, EBOX02 requiert min 38.69" L.';

    // ── Try custom battery path ──────────────────────────────────────────────
    Logger.log('AUCUNE_BATTERIE_COMPATIBLE — attempting custom battery path');
    const compartmentStr = bestMatch.compartment_full || (compartmentL + ' x ' + compartmentW + ' x ' + compartmentH);
    const customChargers = getChargersForVoltage(params.voltage, params.ip54_charger, params.heating_option);
    const ghostResult    = runGhostSims(params, compartmentStr, bestMatch, eboxResult, customChargers);

    if (ghostResult) {
      Logger.log('Custom battery ghost sim PASS — running full efficiency search');
      try {
        // ── 1. Re-run findMaxEfficiency without 80% cap ──────────────────────
        const fullPlan = Object.assign({}, ghostResult.ghostPlan, {
          battery_ah:       ghostResult.achievedAh,
          truck_efficiency: 100,
          battery_type:     'VeloxCustom'
        });
        const chargingEvents = ghostResult.chargingEvents;
        const effSearch  = findMaxEfficiency(fullPlan, chargingEvents, 0, params.target_efficiency || 100);
        const maxEff     = effSearch.maxEfficiency;
        const simResults = effSearch.finalResults;
        const workHours  = calculateWorkHours(chargingEvents, params.shifts, maxEff,
                                              params.shift_duration || 8, params.shift_start_hour);

        Logger.log('Custom battery full sim: eff=' + maxEff + '% work=' + workHours.formatted);

        // Read SOC from sheet immediately — sheet still has last sim's data
        const customSOC = readAllSOC();

        // ── 2. Build allSimResults entry shaped like normal pipeline ─────────
        const shiftDesc = params.shifts + 'x' + (params.shift_duration || 8) + 'h';
        const customSimEntry = {
          passed:            simResults.status === 'PASS',
          battery_ah:        ghostResult.targetAh,
          charger_amps:      ghostResult.chargerAmps,
          charger_model:     ghostResult.chargerModel,
          battery_type:      'VeloxCustom',
          battery_label:     '[CUSTOM] VeloxCustom ' + params.voltage + 'V ' + ghostResult.targetAh + 'Ah',
          ebox:              'CUSTOM',
          max_efficiency:    maxEff,
          work_hours:        workHours.formatted,
          results:           simResults,
          soc_over_time:     customSOC,
          chargingEvents:    chargingEvents,
          truck_load:        fullPlan.truck_load,
          label:             'Sim 1 — ' + ghostResult.achievedAh + 'Ah + ' + ghostResult.chargerAmps + 'A (VeloxCustom)',
          scenario:          'CUSTOM_BATTERY',
          flag:              effSearch.flag || null
        };

        // ── 3. Generate Portée PDF via normal HTML generator ─────────────────
        const porteeParams = Object.assign({}, params, {
          battery_ah:    ghostResult.targetAh,
          charger_amps:  ghostResult.chargerAmps,
          charger_model: ghostResult.chargerModel,
          battery_type:  'VeloxCustom',
          scenario:      'CUSTOM_BATTERY'
        });

        // ── 4. Log to LOGS sheet first — so buildHTMLPortee writes PDF ID to correct row ──
        logSimulationResult(Object.assign({}, porteeParams, {
          battery_ah:       ghostResult.targetAh,
          charger_amps:     ghostResult.chargerAmps,
          charger_model:    ghostResult.chargerModel,
          truck_load:       fullPlan.truck_load,
          truck_efficiency: maxEff,
          work_hours:       workHours.formatted,
          lead_acid_ah:     leadAcidAh,
          compartment:      compartmentStr,
          overall_passed:   simResults.status === 'PASS',
          notes:            '[CUSTOM BATTERY] EVE ' + ghostResult.targetAh + 'Ah — À valider avant envoi dealer',
          flags:            ['CUSTOM_BATTERY_AUTO_GENERATED'],
          scenario:         'CUSTOM_BATTERY'
        }), simResults, 1);

        // ── 5. Generate Portée PDF — now writes PDF ID to the correct last row ──
        const porteePdfBlob = buildHTMLPortee(
          porteeParams,
          [customSimEntry],
          [customSOC]
        );

        // ── 6. Take Data Entry screenshot ────────────────────────────────────
        // Sheet already has max efficiency sim from findMaxEfficiency — just capture
        Utilities.sleep(5000);
        const customDataPdf = capturePdf(porteeParams, fullPlan, 1, '');

        // ── 7. Generate custom battery assessment PDF ────────────────────────
        const customHtml    = generateCustomBatteryHTML(params, ghostResult);
        const customPdfResp = UrlFetchApp.fetch(
          'https://velox-pdf-service-414761682351.us-central1.run.app/generate-pdf',
          {
            method:      'post',
            contentType: 'application/json',
            payload:     JSON.stringify({ html: customHtml, pageFormat: 'Letter' }),
            muteHttpExceptions: true
          }
        );
        const customPdfName = 'VeloxCustom_' + (params.forklift_manufacturer || '') + '_' +
                              (params.forklift_model || '') + '_' + params.voltage + 'V_' + shiftDesc + '.pdf';

        // ── 8. Assemble all PDFs — Portée first, then screenshot, then assessment
        const allCustomPdfs = [];
        if (porteePdfBlob) allCustomPdfs.push(porteePdfBlob);
        if (customDataPdf) allCustomPdfs.push(customDataPdf);
        if (customPdfResp.getResponseCode() === 200) {
          const customPdfBlob = customPdfResp.getBlob().setContentType('application/pdf');
          customPdfBlob.setName(customPdfName);
          allCustomPdfs.push(customPdfBlob);
        } else {
          Logger.log('Custom battery PDF Cloud Run error: ' + customPdfResp.getResponseCode());
        }

        // ── 9. Send via standard pipeline review email ───────────────────────
        const customChargingEvents = buildChargingEvents(
          params.shifts, params.shift_duration || 8, [], 0, params.shift_start_hour
        );
        sendPipelineReviewEmail(
          Object.assign({}, porteeParams, {
            notes: '[CUSTOM BATTERY — À VALIDER] EVE ' + ghostResult.achievedAh + 'Ah — Valider avant envoi dealer'
          }),
          [Object.assign({}, customSimEntry, { simNumber: 1, screenshotIndex: 0 })],
          ['CUSTOM_BATTERY_AUTO_GENERATED'],
          customChargingEvents,
          customChargingEvents,
          allCustomPdfs,
          0,
          60,
          false
        );

        Logger.log('Custom battery: review email sent | Portée + screenshot + assessment attached');

      } catch (e) {
        Logger.log('Custom battery path error: ' + e.message + '\n' + e.stack);
        sendFlagEmail(params, 'AUCUNE_BATTERIE_COMPATIBLE',
          eboxFailDetails + '\n\n[Custom battery path error: ' + e.message + ']', params.thread);
      }
    } else {
      // Ghost sim also failed — send standard flag email
      Logger.log('Custom battery ghost sim also failed — sending flag email');
      sendFlagEmail(params, 'AUCUNE_BATTERIE_COMPATIBLE', eboxFailDetails, params.thread);
    }
    return;
  }
  if (eboxResult.has_avoid_only) pipelineFlags.push('BATTERIES_A_EVITER_SEULEMENT');

  const selection = selectSimulations(dekaResult, eboxResult, params);
  if (selection.stopReason) {
    sendFlagEmail(params, 'SELECTION_IMPOSSIBLE', selection.stopReason, params.thread);
    return;
  }
  selection.flags.forEach(function(f) { pipelineFlags.push(f); });
  Logger.log('Plans: ' + selection.plans.length);

  const is24h = (params.shifts === 3 && params.shift_duration === 8) ||
                (params.shifts === 2 && params.shift_duration === 12);
  if (is24h) pipelineFlags.push('APPLICATION_24H_SOC_VERIFIE_AUX_LIMITES');

  // ── Determine if double-pass is needed ────────────────────────────────────
  const defaultMins  = getDefaultChargeMinutes(params.shifts, params.shift_duration);
  const needsDoubleRun = stated > 0 && stated < defaultMins;

  if (needsDoubleRun) {
    pipelineFlags.push('DOUBLE_SCENARIO_' + stated + '_VS_' + defaultMins + 'MIN — Client a indiqué ' + stated + ' min/shift | Horaire standard: ' + defaultMins + ' min/shift');
    Logger.log('Double-pass: ' + stated + ' min (contrainte) + ' + defaultMins + ' min (standard)');
  }

  // ── Build schedule(s) ─────────────────────────────────────────────────────
const constrainedEvents = buildChargingEvents(params.shifts, params.shift_duration, params.custom_charge_times || [], stated, params.shift_start_hour);
const standardEvents = buildChargingEvents(params.shifts, params.shift_duration, [], 0, params.shift_start_hour);

  // ── Run pass helper ───────────────────────────────────────────────────────
  function runPass(chargingEvents, scenarioLabel, simOffset) {
    let veloxFlexMaxEff = 0;
    const passResults   = [];

    for (let i = 0; i < selection.plans.length; i++) {
      const plan      = selection.plans[i];
      const simNumber = simOffset + i + 1;
      plan.shifts     = params.shifts;

      // ── VeloxPlus gating — check utilization and work threshold ──────────
      if (plan.isVeloxPlusPlan) {
        // Find best VeloxFlex result from this pass
        const flexResults = passResults.filter(function(r) {
          return !String(r.battery_type || '').toUpperCase().includes('PLUS') &&
                 !String(r.battery_type || '').toUpperCase().includes('MAX');
        });

        if (flexResults.length > 0) {
          const bestFlex = flexResults.reduce(function(best, r) {
            return (r.max_efficiency > best.max_efficiency) ? r : best;
          }, flexResults[0]);

          const flexUtil    = bestFlex.results.battery_used_pct || 0;
          const flexPassed  = bestFlex.passed;

          // Skip first Plus plan if Flex passed AND utilization <= 210%
          if (plan.isFirstPlus) {
            const isConstrainedPass = scenarioLabel.includes('CONTRAINTE');
            const bestFlexEff = flexResults.reduce(function(best, r) {
              return (r.max_efficiency > best) ? r.max_efficiency : best;
            }, 0);
            if (flexPassed && flexUtil <= 210 && (!isConstrainedPass || bestFlexEff >= 100)) {
              Logger.log('Skipping VeloxPlus — Flex passed with ' + Math.round(flexUtil) + '% utilization (≤210%)');
              pipelineFlags.push('VELOXPLUS_NON_REQUIS — Flex ' + Math.round(flexUtil) + '% utilization ≤ 210% — VeloxPlus non simulé');
              // Skip all Plus plans by breaking out
              i = selection.plans.length; // jump to end of loop
              break;
            } else {
              const reason = !flexPassed ? 'Flex insuffisant'
                : isConstrainedPass && bestFlexEff < 100 ? 'Contrainte — Flex plafonné à ' + bestFlexEff + '%'
                : 'Utilisation ' + Math.round(flexUtil) + '% > 210%';
              Logger.log('Running VeloxPlus — ' + reason);
              pipelineFlags.push('VELOXPLUS_REQUIS — ' + reason);
            }
          } else {
            // Subsequent Plus plans: skip if previous Plus passed
            const prevPlusResults = passResults.filter(function(r) {
              return String(r.battery_type || '').toUpperCase().includes('PLUS');
            });
           if (prevPlusResults.length > 0 && prevPlusResults[prevPlusResults.length - 1].passed) {
              const prevEff = prevPlusResults[prevPlusResults.length - 1].max_efficiency || 0;
              if (prevEff >= 95) {
                Logger.log('Skipping ' + plan.label + ' — previous VeloxPlus already at ' + prevEff + '%');
                continue;
              }
              Logger.log('Continuing VeloxPlus — previous passed at only ' + prevEff + '% — trying faster charger');
            }
          }
        }
      }

      // ── Legacy conditional skip (for 24h 550/700 skipIf550Passed) ────────
      if (plan.conditional) {
        if (plan.skipIf550Passed) {
          const any550Passed = allSimResults.some(function(r) {
            return r.battery_ah === 550 && r.passed && r.scenario === scenarioLabel;
          });
          if (any550Passed) {
            Logger.log('Skipping ' + plan.label + ' [' + scenarioLabel + '] — 550Ah already passed');
            continue;
          }
        }
      if (plan.skipIfFlexPassed) {
          const prevFlexPassed = passResults.filter(function(r) {
            return !String(r.battery_type || '').toUpperCase().includes('PLUS') &&
                   r.passed && r.scenario === scenarioLabel;
          });
          if (prevFlexPassed.length > 0) {
            const bestFlexEff = prevFlexPassed.reduce(function(best, r) {
              return r.max_efficiency > best ? r.max_efficiency : best;
            }, 0);
            if (bestFlexEff >= 95) {
              Logger.log('Skipping ' + plan.label + ' — previous Flex already at ' + bestFlexEff + '%');
              continue;
            }
            Logger.log('Running ' + plan.label + ' — previous Flex passed at only ' + bestFlexEff + '% — trying faster charger');
          }
        }
      }

      Logger.log('Running ' + plan.label + ' [' + scenarioLabel + ']...');

      const shiftDur = plan.shift_duration || 8;

      // Build shift labels dynamically using shift_start_hour
      const sh = params.shift_start_hour || 7;
      function fmtHour(h) {
        const hNorm = ((h % 24) + 24) % 24;
        if (hNorm === 0) return '12:00 AM';
        if (hNorm === 12) return '12:00 PM';
        if (hNorm < 12) return hNorm + ':00 AM';
        return (hNorm - 12) + ':00 PM';
      }

      if (params.shifts === 3 && shiftDur === 8) {
        plan.shift1_label = '12:00 AM - 8:00 AM';
        plan.shift2_label = '8:00 AM - 4:00 PM';
        plan.shift3_label = '4:00 PM - 11:59 PM';
      } else if (params.shifts === 2 && shiftDur === 12) {
        plan.shift1_label = '12:00 AM - 12:00 PM';
        plan.shift2_label = '12:00 PM - 11:59 PM';
        plan.shift3_label = '';
      } else if (params.shifts === 2 && shiftDur === 8) {
        plan.shift1_label = fmtHour(sh) + ' - ' + fmtHour(sh + 8);
        plan.shift2_label = fmtHour(sh + 8) + ' - ' + fmtHour(sh + 16);
        plan.shift3_label = '';
      } else if (params.shifts === 1 && shiftDur === 12) {
        plan.shift1_label = fmtHour(sh) + ' - ' + fmtHour(sh + 12);
        plan.shift2_label = '';
        plan.shift3_label = '';
      } else {
        plan.shift1_label = fmtHour(sh) + ' - ' + fmtHour(sh + 8);
        plan.shift2_label = '';
        plan.shift3_label = '';
      }

      const effSearch = findMaxEfficiency(plan, chargingEvents, veloxFlexMaxEff, params.target_efficiency || 100);
      const maxEff    = effSearch.maxEfficiency;
      const results   = effSearch.finalResults;

      const isVP = String(plan.battery_type || '').toUpperCase().includes('VELOXPLUS') ||
                   String(plan.battery_type || '').toUpperCase().includes('VELOX PLUS');
      if (!isVP) veloxFlexMaxEff = Math.max(veloxFlexMaxEff, maxEff);

      if (effSearch.flag) pipelineFlags.push('[' + scenarioLabel + '] ' + effSearch.flag);

      const workAtMaxEff = calculateWorkHours(chargingEvents, params.shifts, maxEff, params.shift_duration, params.shift_start_hour);
      const workHours    = workAtMaxEff.formatted;
      results.work_hours = workHours;

      const workPass = meetsMinWorkThreshold(workAtMaxEff.totalMinutes, params.shifts, plan.battery_type, params.shift_duration);
      const ahPass   = results.status === 'PASS';
      const passed   = ahPass && workPass;

     Logger.log(plan.label + ' [' + scenarioLabel + '] | eff: ' + maxEff + '% | work: ' + workHours + ' | pass: ' + passed);

      // ── Bonus 100% verification pass ────────────────────────────────────
      if (passed && maxEff < 100) {
        const nextChargers = (selection.chargers || []).filter(function(c) { return c.amps > plan.charger_amps; });
        let bonus100Found = false;
        for (let nc = 0; nc < nextChargers.length && !bonus100Found; nc++) {
          const bonusCharger = nextChargers[nc];
          const maxSafeAmps = Math.floor(plan.battery_ah * 0.80);
          if (bonusCharger.amps > maxSafeAmps) break;
          const bonusPlan = Object.assign({}, plan, {
            charger_amps:        bonusCharger.amps,
            charger_model:       bonusCharger.model,
            label:               'Sim ' + (simNumber + 1) + ' — ' + plan.battery_ah + 'Ah + ' + bonusCharger.amps + 'A (' + (plan.battery_type || 'VeloxFlex') + ' 100%)',
            simIndex:            simNumber + 1,
            conditional:         false,
            check18KVAThreshold: false
          });
          Logger.log('  Bonus 100% check — ' + plan.battery_ah + 'Ah + ' + bonusCharger.amps + 'A');
          const bonus100Results = runSingleSim(bonusPlan, chargingEvents, 100);
          if (bonus100Results.status === 'PASS' && bonus100Results.min_soc >= 20) {
            const bonusWork     = calculateWorkHours(chargingEvents, params.shifts, 100, params.shift_duration, params.shift_start_hour);
            const bonusWorkPass = meetsMinWorkThreshold(bonusWork.totalMinutes, params.shifts, bonusPlan.battery_type, params.shift_duration);
            bonus100Results.work_hours = bonusWork.formatted;
            Logger.log('  Bonus 100% PASS — ' + bonusCharger.amps + 'A | SOC min: ' + Math.round(bonus100Results.min_soc) + '%');
            runSingleSim(bonusPlan, chargingEvents, 100);
            Utilities.sleep(5000);
            const bonusScheduleTag = needsDoubleRun
              ? (scenarioLabel.includes('CONTRAINTE') ? stated + 'min' : defaultMins + 'min')
              : '';
            allPdfs.push(capturePdf(params, bonusPlan, simNumber + 1, bonusScheduleTag));
            allSimScreenshots.push(readAllSOC());
            const bonusLogParams = {
              dealer: params.dealer || '', customer: params.customer || '',
              sender: params.sender || '', forklift_manufacturer: params.forklift_manufacturer || '',
              forklift_model: params.forklift_model || '', voltage: params.voltage,
              compartment: bestMatch.compartment_full || String(compartmentW),
              battery_ah: bonusPlan.battery_ah, battery_type: bonusPlan.battery_type || 'VeloxFlex',
              charger_amps: bonusPlan.charger_amps, charger_model: bonusPlan.charger_model || '',
              shifts: params.shifts, work_hours: bonusWork.formatted,
              lead_acid_ah: leadAcidAh, truck_load: bonusPlan.truck_load,
              truck_efficiency: 100, use_type: params.use_type,
              heating_option: params.heating_option || false,
              ip54_charger: params.ip54_charger || false,
              gmail_thread_id: params.gmail_thread_id || '',
              email_subject: params.email_subject || '', notes: params.notes || '',
              language: params.language || 'fr',
              scenario: scenarioLabel,
              overall_passed:   bonusWorkPass,
              flags: pipelineFlags.concat(['BONUS_100PCT_VERIFICATION'])
            };
            logSimulationResult(bonusLogParams, bonus100Results, simNumber + 1);
            const bonusSimRecord = {
              simNumber: simNumber + 1, label: bonusPlan.label,
              screenshotIndex: allSimScreenshots.length - 1,
              battery_ah: bonusPlan.battery_ah, battery_type: bonusPlan.battery_type || 'VeloxFlex',
              charger_amps: bonusPlan.charger_amps, charger_model: bonusPlan.charger_model || '',
              charger_model: bonusPlan.charger_model || '', battery_type: bonusPlan.battery_type || '',
              max_efficiency: 100, work_hours: bonusWork.formatted,
              truck_load: bonusPlan.truck_load, compartment_w: bonusPlan.compartment_w || compartmentW,
              ebox: bonusPlan.ebox || '', per_shift: bonusWork.perShift,
              results: bonus100Results, passed: bonusWorkPass,
              scenario: scenarioLabel, chargingEvents: chargingEvents
            };
            passResults.push(bonusSimRecord);
            allSimResults.push(bonusSimRecord);
            bonus100Found = true;
            pipelineFlags.push('BONUS_100PCT — ' + plan.battery_ah + 'Ah + ' + bonusCharger.amps + 'A — SOC min: ' + Math.round(bonus100Results.min_soc) + '%');
          } else {
            Logger.log('  Bonus 100% FAIL — ' + bonusCharger.amps + 'A | SOC min: ' + Math.round(bonus100Results.min_soc || 0) + '%');
          }
        }
      }
      // ── End bonus 100% verification ──────────────────────────────────────

      runSingleSim(plan, chargingEvents, maxEff);

      const scheduleTag = needsDoubleRun
        ? (scenarioLabel.includes('CONTRAINTE') ? stated + 'min' : defaultMins + 'min')
        : '';
      Utilities.sleep(5000); // Let sheet finish rendering before screenshot
allPdfs.push(capturePdf(params, plan, simNumber, scheduleTag));
allSimScreenshots.push(readAllSOC());

      const logParams = {
        dealer:                params.dealer || '',
        customer:              params.customer || '',
        sender:                params.sender || '',
        forklift_manufacturer: params.forklift_manufacturer || '',
        forklift_model:        params.forklift_model || '',
        voltage:               params.voltage,
        compartment:           bestMatch.compartment_full || String(compartmentW),
        battery_ah:            plan.battery_ah,
        battery_type:          plan.battery_type || 'VeloxFlex',
        charger_amps:          plan.charger_amps,
        charger_model:         plan.charger_model || '',
        shifts:                params.shifts,
        work_hours:            workHours,
        lead_acid_ah:          leadAcidAh,
        truck_load:            plan.truck_load,
        truck_efficiency:      maxEff,
        use_type:              params.use_type,
        heating_option:        params.heating_option || false,
        ip54_charger:          params.ip54_charger || false,
        gmail_thread_id:       params.gmail_thread_id || '',
        forklift_class:        params.forklift_class || '',
        special_equipment:     params.special_equipment || '',
        email_subject:         params.email_subject || '',
        notes:                 params.notes || '',
        language:              params.language || 'fr',
        scenario:              scenarioLabel,
        overall_passed:        passed,
        flags:                 pipelineFlags.concat(workPass ? [] : ['HEURES_TRAVAIL_INSUFFISANTES'])
      };
      logSimulationResult(logParams, results, simNumber);

      const simRecord = {
  simNumber:      simNumber,
  screenshotIndex: allSimScreenshots.length - 1,
  label:          plan.label,
  battery_ah:     plan.battery_ah,
  charger_amps:   plan.charger_amps,
  charger_model:  plan.charger_model || '',
  battery_type:   plan.battery_type || '',
  max_efficiency: maxEff,
  work_hours:     workHours,
  truck_load:     plan.truck_load,
  compartment_w:  plan.compartment_w || compartmentW,
  ebox:           plan.ebox || '',
  per_shift:      workAtMaxEff.perShift,
  results:        results,
  passed:         passed,
  scenario:       scenarioLabel,
  chargingEvents: chargingEvents 
};
      passResults.push(simRecord);
      allSimResults.push(simRecord);

      // ── 100% efficiency early exit — skip remaining charger variations ──
      // If this combo already passed at 100% efficiency, no higher charger
      // can improve the result — all work hours would be identical.
      // Only applies when Ebox-constrained (same battery Ah, different chargers).
      if (passed && maxEff >= 100) {
        const isEboxPath = pipelineFlags.some(function(f) {
          return f.includes('EBOX_CONTRAINT_VARIATION_CHARGEUR');
        });
        if (isEboxPath) {
          const remainingPlans = selection.plans.slice(i + 1).filter(function(p) {
            return p.battery_ah === plan.battery_ah;
          });
          if (remainingPlans.length > 0) {
            Logger.log('100% efficiency at ' + plan.charger_amps + 'A — skipping ' +
                       remainingPlans.length + ' remaining charger variation(s) for ' +
                       plan.battery_ah + 'Ah');
            pipelineFlags.push('VARIATION_CHARGEUR_SAUTEE — ' + plan.battery_ah + 'Ah + ' +
                               plan.charger_amps + 'A déjà à 100% — ' +
                               remainingPlans.length + ' variation(s) ignorée(s)');
            break; // exit the for loop — stop processing remaining plans
          }
        }
      }

      // (writeToSimDB now called once after all sims complete)

      // ── 18KVA threshold check — inject Sim 2 dynamically if warranted ──
      if (plan.check18KVAThreshold) {
        const kva18Amps = params.voltage === 36 ? 360
                        : params.voltage === 48 ? 275
                        : params.voltage === 24 ? 400
                        : 0;
        const kva18Charger = kva18Amps > 0 ? (selection.chargers || []).find(function(c) { return c.amps === kva18Amps; }) : null;

        if (kva18Charger) {
          const batteryUsedPct = results.battery_used_pct || 0;
          const overloaded     = batteryUsedPct >= 240;

          if (overloaded) {
            // Battery already overloaded — 18KVA won't help, skip
            pipelineFlags.push('18KVA_NON_TENTE — Utilisation batterie ' + Math.round(batteryUsedPct) + '% >= 240% — chargeur plus rapide n\'aidera pas');
          } else if (!passed) {
            // Sim 1 failed due to insufficient work hours — try 18KVA automatically
            Logger.log('  Flex failed work threshold — injecting 18KVA sim automatically');
            const plan18 = Object.assign({}, plan, {
              charger_amps:        kva18Charger.amps,
              charger_model:       kva18Charger.model,
              label:               'Sim ' + (simNumber + 1) + ' — ' + plan.battery_ah + 'Ah + ' + kva18Charger.amps + 'A (VeloxFlex 18KVA)',
              simIndex:            simNumber + 1,
              conditional:         false,
              check18KVAThreshold: false
            });
            const eff18Search  = findMaxEfficiency(plan18, chargingEvents, veloxFlexMaxEff, params.target_efficiency || 100);
            const maxEff18     = eff18Search.maxEfficiency;
            const results18    = eff18Search.finalResults;
            const work18       = calculateWorkHours(chargingEvents, params.shifts, maxEff18, params.shift_duration, params.shift_start_hour);
            const workPass18   = meetsMinWorkThreshold(work18.totalMinutes, params.shifts, plan18.battery_type, params.shift_duration);
            const passed18     = (results18.status === 'PASS') && workPass18;
            results18.work_hours = work18.formatted;

            Logger.log(plan18.label + ' [' + scenarioLabel + '] | eff: ' + maxEff18 + '% | work: ' + work18.formatted + ' | pass: ' + passed18);
            runSingleSim(plan18, chargingEvents, maxEff18);
            Utilities.sleep(5000);
            allPdfs.push(capturePdf(params, plan18, simNumber + 1, scheduleTag));
            allSimScreenshots.push(readAllSOC());

            const annualHours18  = Math.round((work18.totalMinutes / params.shifts / 60) * params.shifts * 5 * 52);
            const annualHours250 = Math.round((workAtMaxEff.totalMinutes / params.shifts / 60) * params.shifts * 5 * 52);
            const hourGain       = annualHours18 - annualHours250;
            const gainPct        = annualHours250 > 0 ? Math.round((hourGain / annualHours250) * 100) : 0;
            const paybackWeeks   = hourGain > 0 ? Math.round(1000 / (hourGain * 30 / 52)) : null;

            pipelineFlags.push('18KVA_INJECTE_FLEX_INSUFFISANT — ' + kva18Charger.amps + 'A | +' + hourGain + 'h/an (+' + gainPct + '%)' + (paybackWeeks ? ' | Récupération ~' + paybackWeeks + ' sem. à $30/h' : ''));

            const logParams18 = Object.assign({}, {
              dealer: params.dealer || '', customer: params.customer || '',
              sender: params.sender || '', forklift_manufacturer: params.forklift_manufacturer || '',
              forklift_model: params.forklift_model || '', voltage: params.voltage,
              compartment_w: compartmentW, battery_ah: plan18.battery_ah,
              battery_type: plan18.battery_type || 'VeloxFlex',
              charger_amps: plan18.charger_amps, charger_model: plan18.charger_model || '',
              shifts: params.shifts,
              work_hours: work18.formatted, lead_acid_ah: leadAcidAh,
              truck_load: plan18.truck_load, truck_efficiency: maxEff18,
              use_type: params.use_type, heating_option: params.heating_option || false,
              ip54_charger: params.ip54_charger || false, gmail_thread_id: params.gmail_thread_id || '',
              email_subject: params.email_subject || '', notes: params.notes || '',
              language: params.language || 'fr',
              scenario: scenarioLabel,
              overall_passed:   passed18,
              flags: pipelineFlags.concat(workPass18 ? [] : ['HEURES_TRAVAIL_INSUFFISANTES'])
            });
            logSimulationResult(logParams18, results18, simNumber + 1);

            const simRecord18 = {
              simNumber: simNumber + 1, screenshotIndex: allSimScreenshots.length - 1, label: plan18.label,
              battery_ah: plan18.battery_ah, charger_amps: plan18.charger_amps,
              charger_model: plan18.charger_model || '', battery_type: plan18.battery_type || '',
              max_efficiency: maxEff18, work_hours: work18.formatted,
              truck_load: plan18.truck_load, compartment_w: plan18.compartment_w || compartmentW,
              ebox: plan18.ebox || '', per_shift: work18.perShift,
              results: results18, passed: passed18, scenario: scenarioLabel,
              chargingEvents: chargingEvents
            };
            passResults.push(simRecord18);
            allSimResults.push(simRecord18);
            if (!String(plan18.battery_type || '').toUpperCase().includes('PLUS')) {
              veloxFlexMaxEff = Math.max(veloxFlexMaxEff, maxEff18);
            }

          } else if (maxEff >= 100 && (results.battery_used_pct || 0) < 150) {
              pipelineFlags.push('18KVA_NON_TENTE — Flex déjà à 100% avec ' + Math.round(results.battery_used_pct) + '% utilisation — application légère, 18KVA sans bénéfice');
              Logger.log('  18KVA skipped — already 100% efficiency at light utilization');
            } else {
              const annualHours250 = Math.round((workAtMaxEff.totalMinutes / 60) * 5 * 52);
              const threshold6pct  = Math.round(annualHours250 * 0.06);

            // Estimate 18KVA gain: proportional to charger amp ratio
            const ampRatio       = kva18Amps / plan.charger_amps;
            const estimatedGain  = Math.round(annualHours250 * (ampRatio - 1) * 0.15); // conservative estimate

            if (estimatedGain >= threshold6pct) {
              Logger.log('  18KVA threshold met (est. +' + estimatedGain + 'h vs threshold +' + threshold6pct + 'h) — injecting');
              const plan18 = Object.assign({}, plan, {
                charger_amps:        kva18Charger.amps,
                charger_model:       kva18Charger.model,
                label:               'Sim ' + (simNumber + 1) + ' — ' + plan.battery_ah + 'Ah + ' + kva18Charger.amps + 'A (VeloxFlex 18KVA)',
                simIndex:            simNumber + 1,
                conditional:         false,
                check18KVAThreshold: false
              });
              const eff18Search  = findMaxEfficiency(plan18, chargingEvents, veloxFlexMaxEff, params.target_efficiency || 100);
              const maxEff18     = eff18Search.maxEfficiency;
              const results18    = eff18Search.finalResults;
              const work18       = calculateWorkHours(chargingEvents, params.shifts, maxEff18, params.shift_duration, params.shift_start_hour);
              const workPass18   = meetsMinWorkThreshold(work18.totalMinutes, params.shifts, plan18.battery_type, params.shift_duration);
              const passed18     = (results18.status === 'PASS') && workPass18;
              results18.work_hours = work18.formatted;

              Logger.log(plan18.label + ' [' + scenarioLabel + '] | eff: ' + maxEff18 + '% | work: ' + work18.formatted + ' | pass: ' + passed18);
              runSingleSim(plan18, chargingEvents, maxEff18);
              Utilities.sleep(5000);
              allPdfs.push(capturePdf(params, plan18, simNumber + 1, scheduleTag));
              allSimScreenshots.push(readAllSOC());

              const annualHours18 = Math.round((work18.totalMinutes / 60) * 5 * 52);
              const hourGain      = annualHours18 - annualHours250;
              const gainPct       = Math.round((hourGain / annualHours250) * 100);
              const paybackWeeks  = hourGain > 0 ? Math.round(1000 / (hourGain * 30 / 52)) : null;

              pipelineFlags.push('18KVA_SEUIL_ATTEINT — 250A: ' + annualHours250 + 'h/an | 18KVA: ' + annualHours18 + 'h/an | +' + hourGain + 'h (+' + gainPct + '%)' + (paybackWeeks ? ' | Récupération ~' + paybackWeeks + ' sem. à $30/h' : ''));

              const logParams18 = Object.assign({}, {
                dealer: params.dealer || '', customer: params.customer || '',
                sender: params.sender || '', forklift_manufacturer: params.forklift_manufacturer || '',
                forklift_model: params.forklift_model || '', voltage: params.voltage,
                compartment_w: compartmentW, battery_ah: plan18.battery_ah,
                battery_type: plan18.battery_type || 'VeloxFlex',
                charger_amps: plan18.charger_amps, charger_model: plan18.charger_model || '',
                shifts: params.shifts,
                work_hours: work18.formatted, lead_acid_ah: leadAcidAh,
                truck_load: plan18.truck_load, truck_efficiency: maxEff18,
                use_type: params.use_type, heating_option: params.heating_option || false,
                ip54_charger: params.ip54_charger || false, gmail_thread_id: params.gmail_thread_id || '',
                email_subject: params.email_subject || '', notes: params.notes || '',
                scenario: scenarioLabel,
              language: params.language || 'fr',
              overall_passed: passed18,
              flags: pipelineFlags.concat(workPass18 ? [] : ['HEURES_TRAVAIL_INSUFFISANTES'])
              });
              logSimulationResult(logParams18, results18, simNumber + 1);

              const simRecord18 = {
                simNumber: simNumber + 1, label: plan18.label,
                battery_ah: plan18.battery_ah, charger_amps: plan18.charger_amps,
                charger_model: plan18.charger_model || '', battery_type: plan18.battery_type || '',
                max_efficiency: maxEff18, work_hours: work18.formatted,
                truck_load: plan18.truck_load, compartment_w: plan18.compartment_w || compartmentW,
                ebox: plan18.ebox || '', per_shift: work18.perShift,
                results: results18, passed: passed18, scenario: scenarioLabel,
                chargingEvents: chargingEvents
              };
              passResults.push(simRecord18);
              allSimResults.push(simRecord18);
              if (!String(plan18.battery_type || '').toUpperCase().includes('PLUS')) {
                veloxFlexMaxEff = Math.max(veloxFlexMaxEff, maxEff18);
              }

            } else {
              pipelineFlags.push('18KVA_SEUIL_NON_ATTEINT — Gain estimé +' + estimatedGain + 'h/an < seuil 6% (' + threshold6pct + 'h) — 18KVA non simulé');
              Logger.log('  18KVA threshold not met (est. +' + estimatedGain + 'h vs threshold +' + threshold6pct + 'h) — skipping');
            }
          }
        }
      }
    }
    return passResults;
  }

  // ── Execute pass(es) ──────────────────────────────────────────────────────
  const constrainedLabel = needsDoubleRun ? 'CONTRAINTE ' + stated + 'min' : 'STANDARD';
  runPass(constrainedEvents, constrainedLabel, 0);

  if (needsDoubleRun) {
    runPass(standardEvents, 'STANDARD ' + defaultMins + 'min', selection.plans.length);
  }

  // ── Custom weightbox flag — fires when Ebox-constrained and all sims failed ──
  const isEboxConstrained = pipelineFlags.some(function(f) {
    return f.includes('EBOX_CONTRAINT');
  });
  const onlyFailing = allSimResults.length > 0 && allSimResults.every(function(s) {
    return !s.passed;
  });
  if (isEboxConstrained && onlyFailing) {
    pipelineFlags.push(
      'CUSTOM_WEIGHTBOX_A_EVALUER — Aucune solution standard pour compartiment ' +
      compartmentW + '". ' +
      'Batterie sur mesure possible (cellules EVE préférées, CALB en dernier recours). ' +
      'Vérifier outil de layout cellules pour ' + params.forklift_manufacturer + ' ' + params.forklift_model + '.'
    );
  }

  // ── Write all passing combos to SIM_DB once ───────────────────────────────
  const passingSims = allSimResults.filter(function(s) { return s.passed; });
  if (passingSims.length > 0 && params.shifts > 0 && params.shifts <= 3) {
    writeToSimDB(Object.assign({}, params, {
      compartment:  bestMatch.compartment_full || String(compartmentW),
      lead_acid_ah: leadAcidAh
    }), passingSims);
  }

  // Generate Portée PDF and prepend to attachments
const porteePdfBlob = USE_HTML_PDF
  ? buildHTMLPortee(params, allSimResults, allSimScreenshots)
  : buildPorteePDFBlob(params, allSimResults, allSimScreenshots);
if (porteePdfBlob) allPdfs.unshift(porteePdfBlob);

Logger.log('Sending email — ' + allPdfs.length + ' PDFs...');
sendPipelineReviewEmail(params, allSimResults, pipelineFlags, constrainedEvents, standardEvents, allPdfs, stated, defaultMins, needsDoubleRun);
Logger.log('Pipeline complete — ' + allSimResults.length + ' sim(s)');
}

function getDraftTemplate(flagCode, params, dekaMatches) {
  const lang      = params.language || 'fr';
  const firstName = (params.sender_name || '').split(' ')[0] || '';
  const greeting  = firstName
    ? (lang === 'fr' ? 'Salut ' + firstName + ' !' : 'Hey ' + firstName + '!')
    : (lang === 'fr' ? 'Bonjour !' : 'Hi!');
  const mfg    = params.forklift_manufacturer || '';
  const model  = params.forklift_model || '';
  const thanks = lang === 'fr' ? 'Merci!\nMarc-Antoine' : 'Thanks!\nMarc-Antoine';

  if (flagCode === 'CONFIRMATION_REQUISE') {
    let optionsList = '';
    if (dekaMatches && dekaMatches.length > 0) {
      const seenCompartments = new Set();
      let optionIdx = 1;
      dekaMatches.forEach(function(m) {
        if (!seenCompartments.has(m.compartment_full)) {
          seenCompartments.add(m.compartment_full);
          optionsList += '\nOption ' + optionIdx + ' — Caisson: ' + m.compartment_full;
          optionIdx++;
        }
      });
    }
    if (lang === 'fr') {
      return greeting + '\n\nMerci pour ta demande. J\'ai trouvé plusieurs configurations pour le ' +
        mfg + ' ' + model + ' dans notre système:\n' + optionsList +
        '\n\nPeux-tu me confirmer la largeur du caisson de batterie de ton chariot?\n\n' + thanks;
    } else {
      return greeting + '\n\nThanks for your request. I found multiple configurations for the ' +
        mfg + ' ' + model + ' in our system:\n' + optionsList +
        '\n\nCould you confirm the battery compartment width of your forklift?\n\n' + thanks;
    }
  }

  if (flagCode === 'MODELE_INTROUVABLE') {
    if (lang === 'fr') {
      return greeting + '\n\nMerci pour ta demande. Je n\'ai pas trouvé le modèle ' +
        mfg + ' ' + model + ' dans notre base de données.\n\n' +
        'Peux-tu me confirmer:\n' +
        '- Le modèle exact du chariot\n' +
        '- Le voltage (24V, 36V, 48V, 72V ou 80V)\n' +
        '- La largeur du caisson de batterie si disponible\n\n' + thanks;
    } else {
      return greeting + '\n\nThanks for your request. I wasn\'t able to find the ' +
        mfg + ' ' + model + ' in our database.\n\n' +
        'Could you confirm:\n' +
        '- The exact forklift model\n' +
        '- The voltage (24V, 36V, 48V, 72V or 80V)\n' +
        '- The battery compartment width if available\n\n' + thanks;
    }
  }

  if (flagCode === 'MODELE_MANQUANT') {
    if (lang === 'fr') {
      return greeting + '\n\nJ\'aurais besoin de quelques informations pour pouvoir avancer:\n\n' +
        '- Fabricant et modèle du chariot\n' +
        '- Voltage (24V, 36V, 48V, 72V ou 80V)\n' +
        '- Nombre de shifts et durée (ex: 2 shifts de 8h)\n\n' + thanks;
    } else {
      return greeting + '\n\nI\'ll need a few details to move forward:\n\n' +
        '- Forklift manufacturer and model\n' +
        '- Voltage (24V, 36V, 48V, 72V or 80V)\n' +
        '- Number of shifts and duration (e.g. 2 x 8h shifts)\n\n' + thanks;
    }
  }

  return null;
}

  const recipient = 'marc-antoine@veloxpower.com';
  const subject   = '[BLOQUEE] ' + flagCode + ' | ' +
                    (params.forklift_manufacturer || '') + ' ' + (params.forklift_model || '') + ' | ' +
                    (params.customer || params.dealer || 'Client inconnu');

  const body = `Bonjour Marc-Antoine,

Une simulation automatique a ete bloquee.

[ RAISON ]
Code:    ${flagCode}
Detail:  ${reason}

[ INFOS RECUES ]
Dealer:  ${params.dealer || 'Non specifie'}
Client:  ${params.customer || 'Non specifie'}
Chariot: ${params.forklift_manufacturer || ''} ${params.forklift_model || ''}
Voltage: ${params.voltage || 'Non specifie'}
Sujet:   ${params.email_subject || 'Non specifie'}

Intervention manuelle requise.`;

  GmailApp.sendEmail(recipient, subject, body);
  Logger.log('Flag email: ' + flagCode);

  // Apply actionable label to original thread

      'MODELE_MANQUANT':      'VELOX-MODELE-MANQUANT',
      'EXTRACTION_ECHOUEE':   'VELOX-EXTRACTION-ECHOUEE'

function sendPipelineReviewEmail(params, allSimResults, flags, constrainedEvents, standardEvents, allPdfs, statedMins, defaultMins, isDoubleRun) {
  const recipient = 'marc-antoine@veloxpower.com';

  const workingSims = allSimResults.filter(function(s) { return s.passed; });
  const overallPass = workingSims.length > 0;
  const statusTag   = overallPass ? '[OK]' : '[ATTENTION]';
  const shiftDesc   = params.shift_duration === 12 ? params.shifts + 'x12h' : params.shifts + 'x8h';

  const subject = statusTag + ' Simulation a reviser — ' +
                  (params.customer || params.dealer) + ' | ' +
                  params.forklift_manufacturer + ' ' + params.forklift_model + ' ' + params.voltage + 'V | ' + shiftDesc;

  function buildSimBlock(sim) {
    const statusTxt = sim.results.status === 'PASS' ? 'PASS' : 'FAIL';
    const soc1 = sim.results.peak_soc_s1 !== undefined ? '\nSOC max shift 1:     ' + Math.round(sim.results.peak_soc_s1) + '%' : '';
    const soc2 = sim.results.peak_soc_s2 !== undefined ? '\nSOC max shift 2:     ' + Math.round(sim.results.peak_soc_s2) + '%' : '';
    const soc3 = sim.results.peak_soc_s3 !== undefined ? '\nSOC max shift 3:     ' + Math.round(sim.results.peak_soc_s3) + '%' : '';
    return `
────────────────────────────────────────────────
[ SIM ${sim.simNumber} — ${params.voltage}V — ${sim.battery_ah}Ah + ${sim.charger_amps}A — ${sim.battery_type || 'VeloxFlex'} ]
────────────────────────────────────────────────
Batterie:            ${sim.battery_ah} Ah
Chargeur:            ${sim.charger_amps} A${sim.charger_model ? ' (' + sim.charger_model + ')' : ''}
Voltage:             ${params.voltage}V
Compartiment:        ${sim.compartment_w ? sim.compartment_w + '"' : 'Non specifie'}
Ebox:                ${sim.ebox || 'Non specifie'}
Consommation:        ${sim.truck_load} A/hm
Efficacite max:      ${sim.max_efficiency}%
Statut Ah:           ${statusTxt}
Total Ah consomme:   ${Math.round(sim.results.total_ah)} Ah
Ah permis/24h:       ${Math.round(sim.results.allowed_ah)} Ah
Ah restant:          ${Math.round(sim.results.remaining_ah)} Ah
Heures travail:      ${sim.work_hours}
SOC minimum atteint: ${Math.round(sim.results.min_soc)}%${soc1}${soc2}${soc3}
% batterie utilisee: ${Math.round(sim.results.battery_used_pct)}%
Resultat global:     ${sim.passed ? '✅ SUFFISANT' : '❌ INSUFFISANT'}
`;
  }

  function buildScenarioSection(scenarioLabel, headerTitle, headerNote) {
    const sims    = allSimResults.filter(function(s) { return s.scenario === scenarioLabel; });
    const passing = sims.filter(function(s) { return s.passed; });
    const failing = sims.filter(function(s) { return !s.passed; });
    const sep     = '════════════════════════════════════════════════';

    let section = '\n\n' + sep +
                  '\n' + headerTitle +
                  '\n' + headerNote +
                  '\n' + sep;

    if (passing.length > 0) {
      section += '\n\n>>> COMBOS VIABLES — ' + headerTitle + ' (' + passing.length + ') <<<';
      passing.forEach(function(s) { section += '\n' + buildSimBlock(s); });
    }
    if (failing.length > 0) {
      section += '\n\n>>> COMBOS NON VIABLES — ' + headerTitle + ' (' + failing.length + ') <<<';
      failing.forEach(function(s) { section += '\n' + buildSimBlock(s); });
    }
    return section;
  }

  function buildChargeTable(events) {
    return events.map(function(e) {
      return '\n   ' + e.label + ': ' + e.start_time + ' - ' + e.end_time;
    }).join('');
  }

  let scenarioSections = '';
  let chargingSection  = '';

  if (isDoubleRun) {
    const labelA = 'CONTRAINTE ' + statedMins + 'min';
    const labelB = 'STANDARD ' + defaultMins + 'min';
    scenarioSections += buildScenarioSection(
      labelA,
      'SCÉNARIO A — CONTRAINTE CLIENT (' + statedMins + ' MIN/SHIFT)',
      'Horaire basé sur ce que le client a indiqué. Envoyer ces résultats au dealer en premier.'
    );
    scenarioSections += buildScenarioSection(
      labelB,
      'SCÉNARIO B — HORAIRE STANDARD (' + defaultMins + ' MIN/SHIFT)',
      'Si le client peut accorder plus de temps de charge. Utiliser comme levier de négociation si Scénario A est insuffisant.'
    );
    chargingSection =
      '\n\n[ HORAIRE DE CHARGE — SCÉNARIO A (' + statedMins + ' MIN/SHIFT) ]' + buildChargeTable(constrainedEvents) +
      '\n\n[ HORAIRE DE CHARGE — SCÉNARIO B (' + defaultMins + ' MIN/SHIFT) ]' + buildChargeTable(standardEvents);
  } else {
    const simsPass = allSimResults.filter(function(s) { return s.passed; });
    const simsFail = allSimResults.filter(function(s) { return !s.passed; });
   if (simsPass.length > 0) {
      scenarioSections += '\n\n>>> COMBOS VIABLES (' + simsPass.length + ') <<<';
      simsPass.slice().sort(function(a,b) { return a.simNumber - b.simNumber; })
              .forEach(function(s) { scenarioSections += '\n' + buildSimBlock(s); });
    }
    if (simsFail.length > 0) {
      scenarioSections += '\n\n>>> COMBOS NON VIABLES (' + simsFail.length + ') <<<';
      simsFail.slice().sort(function(a,b) { return a.simNumber - b.simNumber; })
              .forEach(function(s) { scenarioSections += '\n' + buildSimBlock(s); });
    }
    chargingSection = '\n\n[ HORAIRE DE CHARGE UTILISE ]' + buildChargeTable(constrainedEvents);
  }

  const shiftLabel = params.shift_duration === 12 ? params.shifts + 'x 12h' : params.shifts + ' shifts';
  const flagsText  = flags.length > 0 ? flags.join('\n   ') : 'Aucun';
  const totalViable = allSimResults.filter(function(s) { return s.passed; }).length;
  const totalNonViable = allSimResults.length - totalViable;

  const body = `Bonjour Marc-Antoine,

${allSimResults.length} simulation(s) — ${totalViable} viable(s), ${totalNonViable} non viable(s).${isDoubleRun ? '\n2 scénarios de charge: Scénario A (' + statedMins + ' min/shift — contrainte client) | Scénario B (' + defaultMins + ' min/shift — horaire standard).' : ''}

[ INFORMATIONS GENERALES ]
Dealer:            ${params.dealer || 'Non specifie'}
Client:            ${params.customer || 'Non specifie'}
Expediteur:        ${params.sender || 'Non specifie'}
Chariot:           ${params.forklift_manufacturer} ${params.forklift_model}
Voltage:           ${params.voltage}V
Shifts:            ${shiftLabel}
Type utilisation:  ${params.use_type === 'heavy' ? 'Lourde' : params.use_type === 'moderate' ? 'Modérée' : 'Moyenne'}
Sujet email:       ${params.email_subject || 'Non specifie'}
${scenarioSections}
${chargingSection}

[ DRAPEAUX / NOTES ]
   ${flagsText}

Simulation generee automatiquement — veuillez reviser avant d'envoyer au dealer.`;

  try {
    const thread = params.thread || (params.gmail_thread_id ? GmailApp.getThreadById(params.gmail_thread_id) : null);
    if (thread) {
      const messages = thread.getMessages();
      // Find last message FROM marc-antoine to reply to — never reply to dealer messages
      let replyTarget = null;
      for (var rm = messages.length - 1; rm >= 0; rm--) {
        if (messages[rm].getFrom().toLowerCase().includes('marc-antoine@veloxpower.com')) {
          replyTarget = messages[rm];
          break;
        }
      }
      if (replyTarget) {
        replyTarget.reply(body, { attachments: allPdfs });
        Logger.log('Review reply sent in thread — ' + totalViable + ' viable, ' + totalNonViable + ' non-viable');
      } else {
        try {
          const lastMessage = messages[messages.length - 1];
const msgData = Gmail.Users.Messages.get('me', lastMessage.getId(), { format: 'metadata', metadataHeaders: ['Message-ID'] });
const rfcHeader = msgData.payload.headers.find(function(h) { return h.name === 'Message-ID' || h.name === 'message-id'; });
const rfcMessageId = rfcHeader ? rfcHeader.value : '<' + lastMessage.getId() + '@mail.gmail.com>';
Logger.log('RFC Message-ID: ' + rfcMessageId);
          GmailApp.sendEmail(
            'marc-antoine@veloxpower.com',
            subject,
            body,
            {
              attachments: allPdfs,
              headers: {
                'In-Reply-To': rfcMessageId,
                'References': rfcMessageId
              }
            }
          );
          Logger.log('Review email threaded via RFC headers — no marc-antoine message existed');
        } catch(e) {
          Logger.log('RFC threading failed, falling back to standalone: ' + e.message);
          GmailApp.sendEmail(recipient, subject, body, { attachments: allPdfs });
        }
      }
    } else {
      GmailApp.sendEmail(recipient, subject, body, { attachments: allPdfs });
      Logger.log('Review email sent (no thread) — ' + totalViable + ' viable, ' + totalNonViable + ' non-viable');
    }
  } catch(e) {
    Logger.log('Thread reply failed, falling back to email: ' + e.message);
    GmailApp.sendEmail(recipient, subject, body, { attachments: allPdfs });
  }
}

// PENDING SIMULATION RETRY
// ─────────────────────────────────────────────────────────────────────────────

function checkPendingSimulations() {
  Logger.log('Checking VELOX-A-SIMULER queue...');
  processQueuedTrucks();

  const pendingLabel   = GmailApp.getUserLabelByName('VELOX-A-SIMULER');
  const processedLabel = GmailApp.getUserLabelByName('VELOX-PROCESSED') ||
                         GmailApp.createLabel('VELOX-PROCESSED');

  if (!pendingLabel) {
    Logger.log('No VELOX-A-SIMULER label found — nothing to retry');
    return;
  }
  const knownDealersRetry = getKnownDealers();

  const threads = GmailApp.search('label:VELOX-A-SIMULER', 0, 10);
  Logger.log('Pending threads to retry: ' + threads.length);

  threads.forEach(function(thread) {
    try {
      const subject    = thread.getFirstMessageSubject();
      const messages   = thread.getMessages();
      // Skip pipeline-generated simulation result emails — keep all human messages
      const humanMessages = messages.filter(function(m) {
        const body = m.getPlainBody();
        return !body.includes('Simulation generee automatiquement — veuillez reviser') &&
               !body.includes('CUSTOM BATTERY — VALIDATION REQUISE') &&
               !body.includes('CUSTOM_BATTERY_AUTO_GENERATED');
      });
      const latestBody = humanMessages.length > 0
        ? humanMessages[humanMessages.length - 1].getPlainBody()
        : messages[messages.length - 1].getPlainBody();
      // ── Extract text from PDF attachments ─────────────────────────────
        let attachmentText = '';
        try {
          const allMessages = thread.getMessages();
          allMessages.forEach(function(msg) {
            const attachments = msg.getAttachments();
            attachments.forEach(function(att) {
              const name = att.getName().toLowerCase();
              if (name.endsWith('.pdf') && !msg.getFrom().toLowerCase().includes('veloxpower.com')) {
                const blob = att.copyBlob();
                blob.setContentType('application/pdf');
               const file = Drive.Files.create(
                  { name: 'tmp_ocr_' + att.getName(), mimeType: 'application/vnd.google-apps.document' },
                  blob,
                  { convert: true }
                );
                const doc  = DocumentApp.openById(file.id);
                const text = doc.getBody().getText();
                attachmentText += '\n\n[ATTACHMENT: ' + att.getName() + ']\n' + text;
                Drive.Files.remove(file.id);
                Logger.log('PDF extracted: ' + att.getName() + ' — ' + text.length + ' chars');
              }
            });
          });
        } catch(e) {
          Logger.log('PDF attachment extraction error: ' + e.message);
        }

        const emailText = subject + '\n\n' + latestBody + attachmentText;

      Logger.log('Retrying: ' + subject);

      let params;
      try {
        params = extractSimulationParams(emailText);
      } catch(e) {
        if (e.message === 'GEMINI_503_EXHAUSTED') {
          Logger.log('Still overloaded — keeping VELOX-A-SIMULER: ' + subject);
        } else {
          thread.removeLabel(pendingLabel);
          thread.addLabel(processedLabel);
          sendFlagEmail(
            { email_subject: subject },
            'EXTRACTION_ECHOUEE',
            'Gemini n\'a pas pu extraire les parametres apres plusieurs tentatives. Traitement manuel requis.',
            thread
          );
          Logger.log('Real failure after retry — flagged: ' + subject);
        }
        return;
      }

     // Extraction succeeded — run pipeline
      thread.removeLabel(pendingLabel);
      thread.addLabel(processedLabel);
      thread.markRead();

params.email_subject   = subject;

      // ── Dealer name normalization ─────────────────────────────────────
      const knownDealersRetry = getKnownDealers();
      params.dealer = normalizeDealer(params.dealer, knownDealersRetry);

      if (!params.voltage || params.voltage === 0) {
  var prescanned = _prescanVoltage(emailText);
  if (prescanned > 0) {
    params.voltage = prescanned;
    Logger.log('Voltage pre-scanned from email: ' + prescanned + 'V');
  }
}
      params.gmail_thread_id = thread.getId();
      params.thread          = thread;

      // Apply dealer and rep labels if they exist
if (params.dealer) {
  const dealerLabel = GmailApp.getUserLabelByName(params.dealer);
  if (dealerLabel) {
    thread.addLabel(dealerLabel);
    Logger.log('Dealer label applied: ' + params.dealer);
  }
  // Apply rep sub-label if it exists (e.g. "Battelec Canada/Patrick McCourt")
  if (params.sender_name) {
    const repLabelName = params.dealer + '/' + params.sender_name;
    const repLabel = GmailApp.getUserLabelByName(repLabelName);
    if (repLabel) {
      thread.addLabel(repLabel);
      Logger.log('Rep label applied: ' + repLabelName);
    }
  }
}

      const hasModel  = params.forklift_manufacturer && params.forklift_model &&
                  String(params.forklift_manufacturer).length > 0 &&
                  String(params.forklift_model).length > 0;
      const hasShifts = params.shifts && params.shifts > 0;

      if (!hasModel) {
        sendFlagEmail(params, 'MODELE_MANQUANT',
          'Fabricant ou modele non detecte. Traitement manuel requis.');
        return;
      }

      if (!hasShifts) {
        // Before firing default runs, check SIM_DB for this client
        const refModel = params.referenced_model || params.forklift_model || '';
        const preCheck = lookupPreviousSimulation(params.dealer, refModel, params.customer);

        if (preCheck && preCheck.match === 'EXACT') {
          Logger.log('SIM_DB pre-check EXACT — using ' + preCheck.data.shifts + ' shifts from DB');
          params.shifts         = preCheck.data.shifts;
          params.shift_duration = preCheck.data.shift_duration || 8;
          params.use_type       = preCheck.data.use_type || 'average';
          runSimulationPipeline(params);
        } else {
          Logger.log('No SIM_DB pre-check hit — running default simulations (2x8h + 3x8h)');
          const params2 = Object.assign({}, params, {
            shifts: 2, shift_duration: 8, use_type: 'average',
            email_subject: subject + ' [AUTO-2x8h]',
            notes: 'Shifts non specifies — simulation par defaut 2x8h'
          });
          runSimulationPipeline(params2);
          const params3 = Object.assign({}, params, {
            shifts: 3, shift_duration: 8, use_type: 'average',
            email_subject: subject + ' [AUTO-3x8h]',
            notes: 'Shifts non specifies — simulation par defaut 3x8h'
          });
          runSimulationPipeline(params3);
        }
      } else {
        if (!params.shift_duration) params.shift_duration = 8;
        if (!params.use_type) params.use_type = 'average';
        runSimulationPipeline(params);
      }

      Logger.log('Pending simulation completed: ' + subject);

    } catch(e) {
      Logger.log('Error processing pending thread: ' + e.toString());
    }
  });

  Logger.log('Pending check complete');
}

function setupPendingTrigger() {
  ScriptApp.getProjectTriggers().forEach(function(t) {
    if (t.getHandlerFunction() === 'checkPendingSimulations') {
      ScriptApp.deleteTrigger(t);
    }
  });
  ScriptApp.newTrigger('checkPendingSimulations')
    .timeBased()
    .everyMinutes(1)
    .create();
  Logger.log('Pending trigger installed — checking every 30 minutes');
}

function removePendingTrigger() {
  ScriptApp.getProjectTriggers().forEach(function(t) {
    if (t.getHandlerFunction() === 'checkPendingSimulations') {
      ScriptApp.deleteTrigger(t);
      Logger.log('Pending trigger removed');
    }
  });
}

function isExternalThread(thread) {
  const messages = thread.getMessages();
  if (messages.length === 0) return false;

  const hasExternalMessage = messages.some(function(m) {
    return !m.getFrom().toLowerCase().includes('veloxpower.com');
  });
  if (!hasExternalMessage) return false;

  const firstSender = messages[0].getFrom().toLowerCase();
  if (!firstSender.includes('veloxpower.com')) return true;

  const subject = thread.getFirstMessageSubject().toUpperCase();
  if (subject.startsWith('FWD:') || subject.startsWith('TR:') ||
      subject.startsWith('FW:') || subject.startsWith('RE:')) return true;

  return false;
}

  Logger.log('Checking for new simulation requests...');

      const threadLabels = thread.getLabels().map(function(l) { return l.getName(); });
      if (threadLabels.includes('VELOX-PROCESSED')) {
        thread.markRead();
        Logger.log('Skipping already-processed thread: ' + thread.getFirstMessageSubject());
        return;
      }
      const isExternal = isExternalThread(thread);

      const trustedInternalSenders = ['kevin@veloxpower.com', 'quote@veloxpower.com', 'carolane@veloxpower.com'];

        return firstSender.includes(s);
      });

        Logger.log('Skipping internal thread: ' + thread.getFirstMessageSubject() + ' | isExternal: ' + isExternal + ' | isTrustedInternal: ' + isTrustedInternal + ' | firstSender: ' + firstSender);
        return;
      }

      const scoreThreshold = isTrustedInternal ? 6 : 6;

      const subject    = thread.getFirstMessageSubject();
      const messages   = thread.getMessages();
      const latestBody = messages[messages.length - 1].getPlainBody();
      const result     = scoreEmail(subject, latestBody, manufacturers);

      if (result.score >= scoreThreshold) {
        Logger.log('MATCH — Score: ' + result.score + ' (threshold: ' + scoreThreshold + ') | Subject: ' + subject);

        thread.addLabel(processedLabel);

                const file     = Drive.Files.create(
                  { name: 'tmp_ocr_' + att.getName(), mimeType: 'application/vnd.google-apps.document' },
                  blob,
                  { convert: true }
                );
                const doc      = DocumentApp.openById(file.id);
                const text     = doc.getBody().getText();
                attachmentText += '\n\n[ATTACHMENT: ' + att.getName() + ']\n' + text;
                // Clean up temp file
                Drive.Files.remove(file.id);
                Logger.log('PDF extracted: ' + att.getName() + ' — ' + text.length + ' chars');
              }
            });
          });
        } catch(e) {
          Logger.log('PDF attachment extraction error: ' + e.message);
        }

        const emailText = subject + '\n\n' + latestBody + attachmentText;
        let params;
        try {
          params = extractSimulationParams(emailText);
        } catch(e) {
          Logger.log('Gemini extraction failed: ' + e.toString());

          if (e.message === 'GEMINI_503_EXHAUSTED') {
            // Temporary overload — label for retry, do NOT send flag email
            const pendingLabel = GmailApp.getUserLabelByName('VELOX-A-SIMULER') ||
                                 GmailApp.createLabel('VELOX-A-SIMULER');
            thread.addLabel(pendingLabel);
            // Remove VELOX-PROCESSED so the pending checker can find it
            thread.removeLabel(processedLabel);
            Logger.log('Gemini overloaded — labeled VELOX-A-SIMULER for retry');
          } else {
            // Real extraction failure — needs manual attention
            sendFlagEmail(
              { email_subject: subject },
              'EXTRACTION_ECHOUEE',
              'Gemini n\'a pas pu extraire les parametres. Traitement manuel requis.'
            );
          }
          return;
        }

params.email_subject   = subject;
        params.gmail_thread_id = thread.getId();
        params.thread          = thread;

        // ── Dealer name normalization ─────────────────────────────────────
        params.dealer = normalizeDealer(params.dealer, knownDealers);

        // ── Multi-truck routing ───────────────────────────────────────────
        if (params.trucks && params.trucks.length > 1) {
          Logger.log('Multi-truck detected: ' + params.trucks.length + ' trucks');
          const sharedParams = {
            dealer:          params.dealer || '',
            customer:        params.customer || '',
            sender_name:     params.sender_name || '',
            sender_email:    params.sender_email || '',
            sender:          [params.sender_name || '', params.sender_email || ''].filter(Boolean).join(' '),
            shifts:          params.shifts || 2,
            shift_duration:  params.shift_duration || 8,
            use_type:        params.use_type || 'average',
            heating_option:  params.heating_option || false,
            ip54_charger:    params.ip54_charger || false,

          };

          params.trucks.forEach(function(truck) {
            const truckParams = Object.assign({}, sharedParams, {
              forklift_manufacturer: truck.forklift_manufacturer || '',
              forklift_model:        truck.forklift_model || '',
              voltage:               truck.voltage || 0,

        // Apply dealer and rep labels if they exist
if (params.dealer) {

    if (repLabel) {
      thread.addLabel(repLabel);
      Logger.log('Rep label applied: ' + repLabelName);
    }
  }
}

        const hasModel  = params.forklift_manufacturer && params.forklift_model &&
                          params.forklift_manufacturer.length > 0 &&
                          params.forklift_model.length > 0;
        const hasShifts = params.shifts && params.shifts > 0;

        if (!hasModel) {
          Logger.log('Missing forklift model — sending flag');
          sendFlagEmail(params, 'MODELE_MANQUANT',
            'Fabricant ou modele non detecte dans l\'email. Traitement manuel requis.');
          return;
        }

        if (!hasShifts) {
          // Before firing default runs, check SIM_DB for this client
          const refModel = params.referenced_model || params.forklift_model || '';
          const preCheck = lookupPreviousSimulation(params.dealer, refModel, params.customer);

          if (preCheck && preCheck.match === 'EXACT') {
            // SIM_DB EXACT hit — use its shift count, fire single run
            Logger.log('SIM_DB pre-check EXACT — using ' + preCheck.data.shifts + ' shifts from DB');
            params.shifts         = preCheck.data.shifts;
            params.shift_duration = preCheck.data.shift_duration || 8;
            params.use_type       = preCheck.data.use_type || 'average';
            runSimulationPipeline(params);
          } else {
            // No DB hit — fire both default runs
            Logger.log('No SIM_DB pre-check hit — running default simulations (2x8h + 3x8h)');
            const params2 = Object.assign({}, params, {
              shifts: 2, shift_duration: 8, use_type: 'average',
              email_subject: subject + ' [AUTO-2x8h]',
              notes: 'Shifts non specifies — simulation par defaut 2x8h'

            const params3 = Object.assign({}, params, {
              shifts: 3, shift_duration: 8, use_type: 'average',
              email_subject: subject + ' [AUTO-3x8h]',
              notes: 'Shifts non specifies — simulation par defaut 3x8h'
            });
            runSimulationPipeline(params3);
          }
        } else {
          if (!params.shift_duration) params.shift_duration = 8;
          if (!params.use_type) params.use_type = 'average';
          runSimulationPipeline(params);
        }

      } else {
        Logger.log('No match — Score: ' + result.score + ' | Subject: ' + subject);
      }

    } catch(e) {
      Logger.log('Error processing thread: ' + e.toString());
    }
  });

  Logger.log('Trigger check complete');
}

function getImageDimensions(fileId) {
  var bytes = DriveApp.getFileById(fileId).getBlob().getBytes();
  var isPNG = (bytes[1] == 80 && bytes[2] == 78 && bytes[3] == 71);
  if (isPNG) {
    var w = ((bytes[16]&0xFF)<<24)|((bytes[17]&0xFF)<<16)|((bytes[18]&0xFF)<<8)|(bytes[19]&0xFF);
    var h = ((bytes[20]&0xFF)<<24)|((bytes[21]&0xFF)<<16)|((bytes[22]&0xFF)<<8)|(bytes[23]&0xFF);
    return { width: w, height: h };
  }
  return null;
}

function sendToDealerFromRow() {

  var activeRow = sheet.getActiveRange().getRow();

  if (activeRow <= 1) {
    SpreadsheetApp.getUi().alert('Please select a data row in the LOGS sheet first.');
    return;
  }

  var heads = sheet.getRange(1, 1, 1, sheet.getLastColumn()).getValues()[0];
  var vals  = sheet.getRange(activeRow, 1, 1, sheet.getLastColumn()).getValues()[0];

  function col(name) {
    var idx = heads.indexOf(name);
    return idx >= 0 ? vals[idx] : '';
  }

 var groupingKey = col('Grouping Key');
  var threadId    = col('Thread Gmail');
  var dealer      = col('Dealer') || '';

  var langue = 'FR';
  var allDataLang = sheet.getDataRange().getValues();
  var headsLang = allDataLang[0];
  var langCol = headsLang.indexOf('Langue Courriel');
  if (langCol < 0) langCol = headsLang.indexOf('Langue courriel');
  var groupingColLang = headsLang.indexOf('Grouping Key');
  if (langCol >= 0 && groupingColLang >= 0 && groupingKey) {
    for (var li = 1; li < allDataLang.length; li++) {
      if (String(allDataLang[li][groupingColLang]) === groupingKey) {
        var rowLang = String(allDataLang[li][langCol] || '').toUpperCase();

    var headerBodySplit = rawMime.indexOf('\r\n\r\n');
    if (headerBodySplit < 0) headerBodySplit = rawMime.indexOf('\n\n');
    var headers = rawMime.substring(0, headerBodySplit);
    var mimeBody = rawMime.substring(headerBodySplit);

    headers = headers.replace(/^(Cc:.*)$/im, function(match) {
      var parts = match.replace(/^Cc:\s*/i, '').split(',');
      var filtered = parts.filter(function(p) {
        return p.toLowerCase().indexOf(marcClean) < 0;
      });
      return filtered.length > 0 ? 'Cc: ' + filtered.join(',') : '';
    });

    headers = headers.replace(/^(To:.*)$/im, function(match) {
      var parts = match.replace(/^To:\s*/i, '').split(',');
      var filtered = parts.filter(function(p) {
        return p.toLowerCase().indexOf(marcClean) < 0;
      });
      return 'To: ' + filtered.join(',');
    });

    // Re-encode and update the draft
    var patchedMime = headers + mimeBody;
    var rawBytes = [];
    for (var ci = 0; ci < patchedMime.length; ci++) {
      rawBytes.push(patchedMime.charCodeAt(ci));
    }
    var encodedMime = Utilities.base64EncodeWebSafe(rawBytes);

    Gmail.Users.Drafts.update({ message: { raw: encodedMime } }, 'me', draftId);
    Logger.log('CC patched — marc-antoine removed from draft');

  } catch(e) {
    Logger.log('CC patch failed (draft still usable): ' + e.message);
  }

function _parseWorkHoursToMinutes(workStr) {
  if (!workStr) return 0;
  var s = String(workStr).trim();
  var hMatch = s.match(/^(\d+)h(\d+)?/i);
  if (hMatch) return parseInt(hMatch[1]) * 60 + parseInt(hMatch[2] || 0);
  var n = parseFloat(s);
  return isNaN(n) ? 0 : Math.round(n * 60);
}

function _esc2(str) {
  return String(str || '').replace(/&/g,'&amp;').replace(/</g,'&lt;').replace(/>/g,'&gt;');
}

function _prescanVoltage(emailText) {
  var text = emailText.toUpperCase();
  var voltages = [96, 80, 72, 48, 36, 24];
  for (var i = 0; i < voltages.length; i++) {
    var v = voltages[i];
    var regex = new RegExp('\\b' + v + '\\s*V\\b', 'i');
    if (regex.test(text)) return v;
  }
  return 0;
}

  const rows = [];

  dealerTabs.forEach(function(tabName) {
    const sheet = roiSS.getSheetByName(tabName);
    if (!sheet) { Logger.log('Tab not found: ' + tabName); return; }

    const data    = sheet.getDataRange().getValues();
    const headers = data[0];
    const colIdx  = function(name) { return headers.indexOf(name); };

    for (var i = 1; i < data.length; i++) {
      const row = data[i];
      const forklift    = String(row[colIdx('Forklift')]     || '').trim();
      const battery     = String(row[colIdx('Battery')]      || '').trim();
      const charger     = String(row[colIdx('Charger')]      || '').trim();
      const application = String(row[colIdx('Application')]  || '').trim();
      const client      = String(row[colIdx('Client')]       || '').trim();
      const date        = row[colIdx('Date')];

      if (!forklift && !battery && !client) continue;
      if (forklift === 'Forklift') continue;

      // ── Parse voltage ─────────────────────────────────────────────────────
      var voltage = 0;
      var voltMatch = (battery + ' ' + forklift).match(/(\d+)\s*[Vv]/);
      if (voltMatch) voltage = parseInt(voltMatch[1]);

      // ── Parse battery Ah and type ─────────────────────────────────────────
      var batteryAh   = 0;
      var batteryType = '';
      var ahMatch = battery.match(/(\d+)\s*[Aa]h/);
      if (ahMatch) batteryAh = parseInt(ahMatch[1]);
      if (battery.toUpperCase().includes('VELOXPLUS') || battery.toUpperCase().includes('VELOX PLUS')) {
        batteryType = 'VeloxPlus';
      } else if (battery.toUpperCase().includes('VELOXFLEX') || battery.toUpperCase().includes('VELOX FLEX')) {
        batteryType = 'VeloxFlex';
      } else if (battery.toUpperCase().includes('VELOXMAX') || battery.toUpperCase().includes('VELOX MAX')) {
        batteryType = 'VeloxMax';
      }

      // ── Parse charger amps ────────────────────────────────────────────────
      var chargerAmps = 0;
      var chargerMatch = charger.match(/(\d+)\s*[Aa]/);
      if (chargerMatch) chargerAmps = parseInt(chargerMatch[1]);

      // ── Parse shifts ──────────────────────────────────────────────────────
      var shifts = 0;
      var shiftMatch = application.match(/(\d+)\s*[Ss]hift/);
      if (shiftMatch) shifts = parseInt(shiftMatch[1]);
      if (shifts === 0) {
        if (application.includes('3')) shifts = 3;
        else if (application.includes('2')) shifts = 2;
        else if (application.includes('1')) shifts = 1;
      }

      // ── Parse shift duration ──────────────────────────────────────────────
      var shiftDuration = 8;
      if (application.includes('12')) shiftDuration = 12;

      // ── Parse use type ────────────────────────────────────────────────────
      var useType = 'average';
      var appLower = application.toLowerCase();
      if (appLower.includes('heavy') || appLower.includes('lourde') ||
          appLower.includes('winery') || appLower.includes('cold') ||
          appLower.includes('paper') || appLower.includes('arterra') ||
          appLower.includes('goodfellow')) {
        useType = 'heavy';
      } else if (appLower.includes('moderate') || appLower.includes('modér')) {
        useType = 'moderate';
      }

      // ── Parse forklift manufacturer and model ─────────────────────────────
      var fabricant = '';
      var modele    = forklift.replace(/\s*\d+[Vv]\s*/, '').trim();
      var brands    = ['LINDE','TOYOTA','CROWN','RAYMOND','HYSTER','YALE','HYUNDAI',
                       'BOBCAT','DOOSAN','JUNGHEINRICH','STILL','MITSUBISHI','CATERPILLAR',
                       'CLARK','HELI','HANGCHA','NOBLELIFT','SANY','UNICARRIERS','WAJAX',
                       'KOMATSU','NISSAN','KALMAR','TCM','BT','ATLET'];
      brands.forEach(function(b) {
        if (forklift.toUpperCase().includes(b)) { fabricant = b; }
      });
      if (fabricant) modele = modele.replace(new RegExp(fabricant, 'i'), '').trim();

      // ── Format date ───────────────────────────────────────────────────────
      var dateStr = '';
      if (date instanceof Date && !isNaN(date)) {
        dateStr = Utilities.formatDate(date, 'America/Toronto', 'yyyy-MM-dd');
      }

      // ── Lookup Lead_Acid_Ah and Compartment_W via lookupDekaPull ─────────
      var leadAcidAh   = '';
      var compartmentW = '';
      if (fabricant && modele) {
        try {
          var dekaResult = lookupDekaPull(fabricant, modele, voltage, {}, dekaData);
          var bestMatch  = (dekaResult.exact_matches  && dekaResult.exact_matches[0])  ||
                           (dekaResult.fuzzy_matches  && dekaResult.fuzzy_matches[0]);
          if (bestMatch) {
            leadAcidAh   = bestMatch.ah             || '';
            compartmentW = bestMatch.compartment_full || '';
          }
        } catch(e) {
          Logger.log('Deka lookup failed for ' + fabricant + ' ' + modele + ': ' + e.message);
        }
      }

      // ── Lookup Charger_Model from pre-loaded Chargeurs data ───────────────
      var chargerModel = '';
      if (chargeursData && chargerAmps > 0 && voltage > 0) {
        try {
          var chHeaders = chargeursData[0];
          var chVoltCol = chHeaders.indexOf('A ' + voltage + 'V');
          if (chVoltCol >= 0) {
            for (var c = 1; c < chargeursData.length; c++) {
              var chAmps = Number(chargeursData[c][chVoltCol]);
              if (chAmps === chargerAmps) {
                chargerModel = String(chargeursData[c][0] || '').trim();
                break;
              }
            }
          }
        } catch(e) {
          Logger.log('Charger model lookup failed: ' + e.message);
        }
      }

      rows.push([
        tabName,        // A - Dealer
        client,         // B - Client
        fabricant,      // C - Fabricant
        modele,         // D - Modèle
        voltage,        // E - Voltage
        shifts,         // F - Shifts
        shiftDuration,  // G - Shift_Duration
        useType,        // H - Use_Type
        batteryAh,      // I - Battery_Ah
        batteryType,    // J - Battery_Type
        chargerAmps,    // K - Charger_Amps
        chargerModel,   // L - Charger_Model
        compartmentW,   // M - Compartment_W
        leadAcidAh,     // N - Lead_Acid_Ah
        'ROI_SHEET',    // O - Source
        dateStr,        // P - Date
        application     // Q - Notes
      ]);
    }
  });

  if (rows.length > 0) {
    const startRow = simDb.getLastRow() + 1;
    simDb.getRange(startRow, 1, rows.length, 17).setValues(rows);
    Logger.log('Migrated ' + rows.length + ' rows to SIM DB');
  } else {
    Logger.log('No rows found to migrate');
  }
}

function lookupPreviousSimulation(dealer, referencedModel, customer) {
  try {
    const ss    = SpreadsheetApp.openById('1Nv6OZ1-3afIrsyZ4E0MtQ7oA4xZ7QDToMqFhhAFAdD4');
    const simDb = ss.getSheetByName('SIM DB');
    const data  = simDb.getDataRange().getValues();
    const heads = data[0];

    const col = function(name) { return heads.indexOf(name); };
    const clientCol    = col('Client');
    const modelCol     = col('Modèle');
    const fabricantCol = col('Fabricant');
    const voltageCol   = col('Voltage');
    const shiftsCol    = col('Shifts');
    const durCol       = col('Shift_Duration');
    const useCol       = col('Use_Type');
    const battAhCol    = col('Battery_Ah');
    const battTypeCol  = col('Battery_Type');
    const chargerCol   = col('Charger_Amps');
    const chargerModelCol = col('Charger_Model');
    const compCol      = col('Compartment');
    const laCol           = col('Lead_Acid_Ah');

    const modelClean    = (referencedModel || '').toUpperCase().replace(/[^A-Z0-9]/g, '');
    const customerClean = (customer || '').toUpperCase();

    let exactMatch    = null; // same client + same model
    let exactMatchRow = -1;
    let clientMatch   = null; // same client only

    // Search from bottom up — most recent match wins
    for (var i = data.length - 1; i >= 1; i--) {
      const row         = data[i];
      const rowClient   = String(row[clientCol]   || '').toUpperCase();
      const rowModel    = String(row[modelCol]     || '').toUpperCase().replace(/[^A-Z0-9]/g, '');
      const rowFabricant = String(row[fabricantCol] || '').toUpperCase().replace(/[^A-Z0-9]/g, '');
      const rowFull     = rowFabricant + rowModel;

      // Client match — partial is fine
      const clientHit = customerClean.length > 3 &&
                        (rowClient.includes(customerClean) || customerClean.includes(rowClient));

      if (!clientHit) continue;

      const result = {
        dealer:         row[col('Dealer')]  || dealer,
        fabricant:      row[fabricantCol]   || '',
        model:          row[modelCol]       || '',
        voltage:        row[voltageCol]     || 0,
        shifts:         row[shiftsCol]      || 0,
        shift_duration: row[durCol]         || 8,
        use_type:       row[useCol]         || 'average',
        battery_ah:     row[battAhCol]      || 0,
        battery_type:   row[battTypeCol]    || '',
        charger_amps:   row[chargerCol]        || 0,
        charger_model:  row[chargerModelCol]   || '',
        compartment_w:  row[compCol]           || '',  // kept as compartment_w for pipeline compatibility
        lead_acid_ah:   row[laCol]             || 0
      };

      // Check for exact model match
      const modelHit = modelClean.length > 0 &&
                       (rowModel.includes(modelClean) || rowFull.includes(modelClean));

      if (modelHit && !exactMatch) {
  var rawDate = row[col('Date')];
var entryDate = rawDate instanceof Date ? rawDate : new Date(String(rawDate).replace(' a.m.', ' AM').replace(' p.m.', ' PM'));
var ageMinutes = isNaN(entryDate) ? 9999 : (new Date() - entryDate) / 1000 / 60;
  if (ageMinutes < 10) {
    Logger.log('SIM_DB exact match skipped — written ' + Math.round(ageMinutes) + ' min ago (current execution)');
    continue;
  }
  exactMatch = result;
  exactMatchRow = i;
        Logger.log('SIM_DB exact match: ' + row[fabricantCol] + ' ' + row[modelCol] + ' | ' + row[clientCol]);
      } else if (!clientMatch) {
        clientMatch = result;
        Logger.log('SIM_DB client match: ' + row[clientCol] + ' (different model)');
      }

      if (exactMatch) break; // exact match found, stop searching
    }

    if (exactMatch)  return { match: 'EXACT',  data: exactMatch, rowIndex: exactMatchRow };
    if (clientMatch) return { match: 'CLIENT', data: clientMatch };
    return null;

  } catch(e) {
    Logger.log('lookupPreviousSimulation error: ' + e.message);
    return null;
  }
}

function writeToSimDB(params, simResults, porteePdfId) {
  try {
    const ss    = SpreadsheetApp.openById('1Nv6OZ1-3afIrsyZ4E0MtQ7oA4xZ7QDToMqFhhAFAdD4');
    const simDb = ss.getSheetByName('SIM DB');
    const now = new Date();
const today = now.toLocaleString('en-CA', { timeZone: 'America/Toronto' }).replace(',', '');

    // Collapse all passing sim combos into comma-separated strings
    const sims = Array.isArray(simResults) ? simResults : [simResults];
    const uniqueBattAh       = [...new Set(sims.map(s => s.battery_ah))].join(', ');
    const uniqueBattType     = [...new Set(sims.map(s => s.battery_type || 'VeloxFlex'))].join(', ');
    const uniqueChargerAmps  = [...new Set(sims.map(s => s.charger_amps))].join(', ');
    const uniqueChargerModel = [...new Set(sims.map(s => s.charger_model || '').filter(Boolean))].join(', ');

    // Best efficiency across all passing sims
    const bestEff  = Math.max(...sims.map(s => s.efficiency_pct || 0));
    const bestWork = Math.max(...sims.map(s => s.work_hours || 0));

    // Charge constraint
    const chargeConstraint = (params.stated_charge_minutes_per_shift && 
                              params.stated_charge_minutes_per_shift < 60) 
                              ? 'CONTRAINTE' : 'STANDARD';

    const newRow = [
      today,                               // A - Date
      params.dealer                || '',  // B - Dealer
      params.customer              || '',  // C - Client
      params.forklift_manufacturer || '',  // D - Fabricant
      params.forklift_model        || '',  // E - Modèle
      params.forklift_count        || '',  // F - Forklift_Count
      params.voltage               || 0,   // G - Voltage
      params.shifts                || 0,   // H - Shifts
      params.shift_duration        || 8,   // I - Shift_Duration
      params.use_type              || '',  // J - Use_Type
      params.lead_acid_ah          || 0,   // K - Lead_Acid_Ah
      uniqueBattAh,                        // L - Battery_Ah
      uniqueBattType,                      // M - Battery_Type
      uniqueChargerAmps,                   // N - Charger_Amps
      uniqueChargerModel,                  // O - Charger_Model
      params.compartment           || '',  // P - Compartment
      bestEff  || '',                      // Q - Efficiency_%
      bestWork || '',                      // R - Work_Hours
      params.stated_charge_minutes_per_shift || '', // S - Charge_Minutes
      chargeConstraint,                    // T - Charge_Constraint
      'PIPELINE',                          // U - Source
      params.language              || '',  // V - Language
      'ENVOYÉE',                           // W - Status
      'TBD',                               // X - Outcome
      '',                                  // Y - Sold_Combo
      params.sender_name           || '',  // Z - Dealer_Contact
      params.sender_email          || '',  // AA - Dealer_Email
      params.gmail_thread_id       || '',  // AB - Gmail_Thread_ID
      params.email_subject         || '',  // AC - Email_Subject
      porteePdfId                  || '',  // AD - Portée_PDF_ID
      ''                                   // AE - Notes
    ];

    // Dedup key — tighter than before
    const dedupKey = [
      String(params.dealer          || '').trim().toLowerCase(),
      String(params.customer        || '').trim().toLowerCase(),
      String(params.forklift_model  || '').trim().toLowerCase(),
      String(params.voltage         || 0),
      String(params.shifts          || 0),
      String(params.shift_duration  || 8),
      chargeConstraint
    ].join('|');

    if (params._simDbRowUsed >= 0) {
      const sheetRow = params._simDbRowUsed + 1;
      simDb.getRange(sheetRow, 1, 1, newRow.length).setValues([newRow]);
      Logger.log('SIM_DB: updated row ' + sheetRow + ' — ' + params.forklift_model);
    } else {
      const existing = simDb.getDataRange().getValues();
      const heads    = existing[0];
      const dCol     = heads.indexOf('Dealer');
      const cCol     = heads.indexOf('Client');
      const mCol     = heads.indexOf('Modèle');
      const vCol     = heads.indexOf('Voltage');
      const sCol     = heads.indexOf('Shifts');
      const sdCol    = heads.indexOf('Shift_Duration');
      const ccCol    = heads.indexOf('Charge_Constraint');

      let existingRowIdx = -1;
      for (let i = 1; i < existing.length; i++) {
        const row = existing[i];
        const key = [
          String(row[dCol]  || '').trim().toLowerCase(),
          String(row[cCol]  || '').trim().toLowerCase(),
          String(row[mCol]  || '').trim().toLowerCase(),
          String(row[vCol]  || 0),
          String(row[sCol]  || 0),
          String(row[sdCol] || 8),
          String(row[ccCol] || 'STANDARD')
        ].join('|');
        if (key === dedupKey) { existingRowIdx = i; break; }
      }

      if (existingRowIdx >= 0) {
        simDb.getRange(existingRowIdx + 1, 1, 1, newRow.length).setValues([newRow]);
        Logger.log('SIM_DB: updated existing row ' + (existingRowIdx + 1) + ' — ' + params.forklift_model);
      } else {
        simDb.appendRow(newRow);
        Logger.log('SIM_DB: new entry — ' + params.forklift_model + ' | ' + (params.customer || ''));
      }
    }
  } catch(e) {
    Logger.log('writeToSimDB error: ' + e.message);
  }
}

function migrateSIMDB() {
  const ss    = SpreadsheetApp.openById('1Nv6OZ1-3afIrsyZ4E0MtQ7oA4xZ7QDToMqFhhAFAdD4');
  const simDb = ss.getSheetByName('SIM DB');

  // Read existing data
  const existing = simDb.getDataRange().getValues();
  const oldHeaders = existing[0];
  const oldRows = existing.slice(1);

  Logger.log('Migration: found ' + oldRows.length + ' existing rows');

  // New headers
  const newHeaders = [
    'Date', 'Dealer', 'Client', 'Fabricant', 'Modèle', 'Forklift_Count',
    'Voltage', 'Shifts', 'Shift_Duration', 'Use_Type', 'Lead_Acid_Ah',
    'Battery_Ah', 'Battery_Type', 'Charger_Amps', 'Charger_Model',
    'Compartment', 'Efficiency_%', 'Work_Hours', 'Charge_Minutes',
    'Charge_Constraint', 'Source', 'Language', 'Status', 'Outcome',
    'Sold_Combo', 'Dealer_Contact', 'Dealer_Email', 'Gmail_Thread_ID',
    'Email_Subject', 'Portée_PDF_ID', 'Notes'
  ];

  // Map old column indexes (based on current writeToSimDB order)
  // Old: A=Dealer, B=Client, C=Fabricant, D=Modèle, E=Voltage, F=Shifts,
  //      G=Shift_Duration, H=Use_Type, I=Battery_Ah, J=Battery_Type,
  //      K=Charger_Amps, L=Charger_Model, M=Compartment, N=Lead_Acid_Ah,
  //      O=Source, P=Date, Q=Notes(Email_Subject)

  // Deduplicate — keep most recent per dealer+client+model+voltage+shifts+shift_duration
  const deduped = {};
  oldRows.forEach(function(row) {
    if (!row[0] && !row[1] && !row[3]) return; // skip empty rows
    const key = [
      String(row[0]).trim().toLowerCase(), // dealer
      String(row[1]).trim().toLowerCase(), // client
      String(row[3]).trim().toLowerCase(), // model
      String(row[4]),                       // voltage
      String(row[5]),                       // shifts
      String(row[6])                        // shift_duration
    ].join('|');

    // Keep most recent (compare date at col P = index 15)
    if (!deduped[key]) {
      deduped[key] = row;
    } else {
      const existingDate = new Date(deduped[key][15] || 0);
      const newDate      = new Date(row[15] || 0);
      if (newDate > existingDate) deduped[key] = row;
    }
  });

  const dedupedRows = Object.values(deduped);
  Logger.log('Migration: kept ' + dedupedRows.length + ' rows after dedup (' + (oldRows.length - dedupedRows.length) + ' removed)');

  // Map to new 31-column structure
  const newRows = dedupedRows.map(function(row) {
    return [
      row[15] || '',   // A - Date (was col P)
      row[0]  || '',   // B - Dealer
      row[1]  || '',   // C - Client
      row[2]  || '',   // D - Fabricant
      row[3]  || '',   // E - Modèle
      '',              // F - Forklift_Count (new — blank)
      row[4]  || '',   // G - Voltage
      row[5]  || '',   // H - Shifts
      row[6]  || '',   // I - Shift_Duration
      row[7]  || '',   // J - Use_Type
      row[13] || '',   // K - Lead_Acid_Ah (was col N)
      row[8]  || '',   // L - Battery_Ah
      row[9]  || '',   // M - Battery_Type
      row[10] || '',   // N - Charger_Amps
      row[11] || '',   // O - Charger_Model
      row[12] || '',   // P - Compartment
      '',              // Q - Efficiency_% (new — blank)
      '',              // R - Work_Hours (new — blank)
      '',              // S - Charge_Minutes (new — blank)
      'STANDARD',      // T - Charge_Constraint (default)
      row[14] || '',   // U - Source (was col O)
      '',              // V - Language (new — blank)
      'ENVOYÉE',       // W - Status (default)
      'TBD',           // X - Outcome (default)
      '',              // Y - Sold_Combo (new — blank)
      '',              // Z - Dealer_Contact (new — blank)
      '',              // AA - Dealer_Email (new — blank)
      '',              // AB - Gmail_Thread_ID (new — blank)
      row[16] || '',   // AC - Email_Subject (was col Q)
      '',              // AD - Portée_PDF_ID (new — blank)
      ''               // AE - Notes
    ];
  });

  // Clear and rewrite
  simDb.clearContents();
  simDb.getRange(1, 1, 1, newHeaders.length).setValues([newHeaders]);
  if (newRows.length > 0) {
    simDb.getRange(2, 1, newRows.length, newHeaders.length).setValues(newRows);
  }

  // Format header row
  simDb.getRange(1, 1, 1, newHeaders.length).setFontWeight('bold');

  Logger.log('Migration complete — ' + newRows.length + ' rows written with 31-column structure');
}

function processQueuedTrucks() {
  const ss         = SpreadsheetApp.openById('1Nv6OZ1-3afIrsyZ4E0MtQ7oA4xZ7QDToMqFhhAFAdD4');
  const queueSheet = ss.getSheetByName('QUEUE');
  if (!queueSheet) {
    Logger.log('No QUEUE sheet found — nothing to process');
    return;
  }
  const data    = queueSheet.getDataRange().getValues();
  const headers = data[0];
  const statusCol     = headers.indexOf('Status');
  const paramsCol     = headers.indexOf('Params JSON');
  const threadIdCol   = headers.indexOf('Thread ID');

  // Find first PENDING row
  for (let i = 1; i < data.length; i++) {
    if (data[i][statusCol] === 'PENDING') {
      // Mark as IN_PROGRESS immediately to prevent double processing
      queueSheet.getRange(i + 1, statusCol + 1).setValue('IN_PROGRESS');
      SpreadsheetApp.flush();

      try {
        const truckParams = JSON.parse(data[i][paramsCol]);
        const threadId    = data[i][threadIdCol];

        Logger.log('Processing queued truck: ' + truckParams.forklift_manufacturer + ' ' + truckParams.forklift_model);

        // Reattach thread if available
       if (threadId) {
          try {
            truckParams.thread = GmailApp.getThreadById(threadId);
            truckParams.gmail_thread_id = threadId;
          } catch(e) {
            Logger.log('Could not reattach thread: ' + e.message);
          }
        }

       if (!truckParams.shifts || truckParams.shifts === 0) {
          const params2 = Object.assign({}, truckParams, {
            shifts: 2, shift_duration: 8,
            _skipSimDb: true,
            email_subject: (truckParams.email_subject || '') + ' [AUTO2x8h]'
          });
          runSimulationPipeline(params2);
          const params3 = Object.assign({}, truckParams, {
            shifts: 3, shift_duration: 8,
            _skipSimDb: true,
            email_subject: (truckParams.email_subject || '') + ' [AUTO3x8h]'
          });
          runSimulationPipeline(params3);
        } else {
          runSimulationPipeline(truckParams);
        }

        // Mark as DONE
        queueSheet.getRange(i + 1, statusCol + 1).setValue('DONE');
        Logger.log('Queue item completed: row ' + (i + 1));

      } catch(e) {
        queueSheet.getRange(i + 1, statusCol + 1).setValue('ERROR: ' + e.message.substring(0, 50));
        Logger.log('Queue item failed: ' + e.message);
      }

      // Only process ONE truck per run to stay under 6 minutes
      return;
    }
  }
  Logger.log('No PENDING items in QUEUE');
}

function repopulateSIMDBFromLogs() {
  const ss     = SpreadsheetApp.openById('1Nv6OZ1-3afIrsyZ4E0MtQ7oA4xZ7QDToMqFhhAFAdD4');
  const logs   = ss.getSheetByName('LOGS');
  const simDb  = ss.getSheetByName('SIM DB');
  const chargeurs = ss.getSheetByName('Chargeurs');

  // Build charger lookup map: voltage|amps → PN
  const chargeursData = chargeurs.getDataRange().getValues();
  const chargeursHead = chargeursData[0];
  const voltageColMap = {
    24: chargeursHead.indexOf('A 24V'),
    36: chargeursHead.indexOf('A 36V'),
    48: chargeursHead.indexOf('A 48V'),
    72: chargeursHead.indexOf('A 72V'),
    80: chargeursHead.indexOf('A 80V'),
    96: chargeursHead.indexOf('A 96V')
  };

  const chargerLookup = {};
  chargeursData.slice(1).forEach(function(row) {
    const pn = String(row[0] || '').trim();
    if (!pn) return;
    Object.keys(voltageColMap).forEach(function(volt) {
      const col = voltageColMap[volt];
      if (col < 0) return;
      const amps = Number(row[col] || 0);
      if (amps > 0) {
        const key = volt + '|' + amps;
        if (!chargerLookup[key]) chargerLookup[key] = pn;
      }
    });
  });

  Logger.log('Charger lookup built: ' + Object.keys(chargerLookup).length + ' entries');

  const logData = logs.getDataRange().getValues();
  const heads   = logData[0];
  const rows    = logData.slice(1);

  // LOGS column indexes
  const iDate      = heads.indexOf('Date/Heure');
  const iDealer    = heads.indexOf('Dealer');
  const iClient    = heads.indexOf('Client');
  const iSender    = heads.indexOf('Expéditeur');
  const iMfg       = heads.indexOf('Fabricant');
  const iModel     = heads.indexOf('Modèle');
  const iVoltage   = heads.indexOf('Voltage');
  const iComp      = heads.indexOf('Compart. W (po)');
  const iBattType  = heads.indexOf('Battery_Type');
  const iCharModel = heads.indexOf('Charger_Model');
  const iAh        = heads.indexOf('Ah Lithium');
  const iAmps      = heads.indexOf('Ampères');
  const iShifts    = heads.indexOf('Shifts');
  const iWork      = heads.indexOf('Hrs Travail/Shift');
  const iEff       = heads.indexOf('Efficacité %');
  const iStatus    = heads.indexOf('Statut');
  const iUseType   = heads.indexOf('Use Type');
  const iLA        = heads.indexOf('Ah Réf. LA');
  const iTruckLoad = heads.indexOf('Truck Load A/hm');
  const iThread    = heads.indexOf('Thread Gmail');
  const iNotes     = heads.indexOf('Notes/Flags');
  const iSubject   = heads.indexOf('Sujet Email');
  const iScenario  = heads.indexOf('Scénario');
  const iLang      = heads.indexOf('Langue courriel');
  const iPdfId     = heads.indexOf('Portée PDF ID');
  const iGroupKey  = heads.indexOf('Grouping Key');

  // Filter: only PASS or REVIEW with a grouping key
  const validRows = rows.filter(function(row) {
    const status   = String(row[iStatus]   || '');
    const groupKey = String(row[iGroupKey] || '');
    return (status === 'PASS' || status === 'REVIEW') && groupKey !== '';
  });

  Logger.log('repopulate: ' + validRows.length + ' valid LOGS rows found');

  // Group by grouping key + model + shifts + scenario
  const groups = {};
  validRows.forEach(function(row) {
    const groupKey = String(row[iGroupKey] || '');
    const model    = String(row[iModel]    || '');
    const shifts   = String(row[iShifts]   || '');
    const scenario = String(row[iScenario] || 'STANDARD');
    const key      = groupKey + '|' + model + '|' + shifts + '|' + scenario;

    if (!groups[key]) {
      groups[key] = {
        date:       row[iDate],
        dealer:     row[iDealer],
        client:     row[iClient],
        sender:     row[iSender],
        mfg:        row[iMfg],
        model:      model,
        voltage:    row[iVoltage],
        comp:       row[iComp],
        shifts:     row[iShifts],
        work:       row[iWork],
        useType:    row[iUseType],
        la:         row[iLA],
        thread:     row[iThread],
        subject:    row[iSubject],
        scenario:   scenario,
        lang:       row[iLang],
        pdfId:      row[iPdfId] || '',
        battAhs:    [],
        battTypes:  [],
        charAmps:   [],
        charModels: [],
        effs:       []
      };
    }

    // Infer Battery_Type from dedicated column or Notes
    const battTypeRaw = String(row[iBattType] || '');
    const notes       = String(row[iNotes]    || '').toUpperCase();
    const inferredBattType = battTypeRaw
      ? battTypeRaw
      : (notes.includes('VELOXPLUS') || notes.includes('VELOX PLUS'))
        ? 'VeloxPlus' : 'VeloxFlex';

    // Infer Charger_Model from dedicated column or lookup
    const charModelRaw = String(row[iCharModel] || '');
    const voltage      = Number(row[iVoltage]   || 0);
    const amps         = Number(row[iAmps]       || 0);
    const lookupKey    = voltage + '|' + amps;
    const inferredCharModel = charModelRaw
      ? charModelRaw
      : (chargerLookup[lookupKey] || '');

    groups[key].battAhs.push(String(row[iAh] || ''));
    groups[key].battTypes.push(inferredBattType);
    groups[key].charAmps.push(String(amps || ''));
    groups[key].charModels.push(inferredCharModel);
    groups[key].effs.push(Number(row[iEff] || 0));

    // Keep latest PDF ID
    if (row[iPdfId]) groups[key].pdfId = row[iPdfId];
  });

  Logger.log('repopulate: ' + Object.keys(groups).length + ' unique groups found');

  // Deduplicate — keep most recent per dealer+client+model+voltage+shifts+scenario
  const deduped = {};
  Object.values(groups).forEach(function(g) {
    const dedupKey = [
      String(g.dealer  || '').trim().toLowerCase(),
      String(g.client  || '').trim().toLowerCase(),
      String(g.model   || '').trim().toLowerCase(),
      String(g.voltage || ''),
      String(g.shifts  || ''),
      String(g.scenario || 'STANDARD')
    ].join('|');

    if (!deduped[dedupKey]) {
      deduped[dedupKey] = g;
    } else {
      const existDate = new Date(deduped[dedupKey].date || 0);
      const newDate   = new Date(g.date || 0);
      if (newDate > existDate) deduped[dedupKey] = g;
    }
  });

  Logger.log('repopulate: ' + Object.keys(deduped).length + ' rows after dedup');

  // Build SIM DB rows
  const newRows = Object.values(deduped).map(function(g) {
    const uniqueBattAh   = [...new Set(g.battAhs)].join(', ');
    const uniqueBattType = [...new Set(g.battTypes)].join(', ');
    const uniqueCharAmps = [...new Set(g.charAmps)].join(', ');
    const uniqueCharMod  = [...new Set(g.charModels.filter(Boolean))].join(', ');
    const bestEff        = Math.max(...g.effs);

    // Extract sender name and email
    const senderRaw   = String(g.sender || '');
    const emailMatch  = senderRaw.match(/[\w.-]+@[\w.-]+\.\w+/);
    const senderEmail = emailMatch ? emailMatch[0] : '';
    const senderName  = senderRaw.replace(senderEmail, '').trim();

    const chargeConstraint = String(g.scenario || '').includes('CONTRAINTE')
                             ? 'CONTRAINTE' : 'STANDARD';
    const chargeMins = String(g.scenario || '').includes('45min') ? 45
                     : String(g.scenario || '').includes('60min') ? 60
                     : 60;

    return [
      Utilities.formatDate(new Date(g.date), 'America/Toronto', 'yyyy-MM-dd'), // A - Date
      g.dealer    || '',   // B - Dealer
      g.client    || '',   // C - Client
      g.mfg       || '',   // D - Fabricant
      g.model     || '',   // E - Modèle
      '',                  // F - Forklift_Count
      g.voltage   || '',   // G - Voltage
      g.shifts    || '',   // H - Shifts
      8,                   // I - Shift_Duration
      g.useType   || '',   // J - Use_Type
      g.la        || '',   // K - Lead_Acid_Ah
      uniqueBattAh,        // L - Battery_Ah
      uniqueBattType,      // M - Battery_Type
      uniqueCharAmps,      // N - Charger_Amps
      uniqueCharMod,       // O - Charger_Model
      g.comp      || '',   // P - Compartment
      bestEff     || '',   // Q - Efficiency_%
      g.work      || '',   // R - Work_Hours
      chargeMins,          // S - Charge_Minutes
      chargeConstraint,    // T - Charge_Constraint
      'PIPELINE',          // U - Source
      g.lang      || '',   // V - Language
      'ENVOYÉE',           // W - Status
      'TBD',               // X - Outcome
      '',                  // Y - Sold_Combo
      senderName,          // Z - Dealer_Contact
      senderEmail,         // AA - Dealer_Email
      g.thread    || '',   // AB - Gmail_Thread_ID
      g.subject   || '',   // AC - Email_Subject
      g.pdfId     || '',   // AD - Portée_PDF_ID
      ''                   // AE - Notes
    ];
  });

  // Append to SIM DB
  if (newRows.length > 0) {
    simDb.getRange(simDb.getLastRow() + 1, 1, newRows.length, 31).setValues(newRows);
    Logger.log('repopulate: ' + newRows.length + ' rows written to SIM DB');
  } else {
    Logger.log('repopulate: no rows to write');
  }
}
